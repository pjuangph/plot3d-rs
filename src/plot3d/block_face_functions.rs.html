44    ///
45    /// * `x`, `y`, `z` - Cartesian coordinates.

47    pub fn add_vertex(&mut self, x: f64, y: f64, z: f64, i: usize, j: usize, k: usize) {
48        self.vertices.push([x, y, z]);
49        self.indices.push([i, j, k]);
50        let n = self.vertices.len() as f64;
51        let mut cx = 0.0;
52        let mut cy = 0.0;
53        let mut cz = 0.0;
54        for v in &self.vertices {
55            cx += v[0];
56            cy += v[1];
57            cz += v[2];
58        }
59        self.centroid = [cx / n, cy / n, cz / n];
60    }

142    /// Determine which index is constant, if the face is structured.
143    pub fn const_axis(&self) -> Option<FaceAxis> {
144        let i_same = self.imin() == self.imax();
145        let j_same = self.jmin() == self.jmax();
146        let k_same = self.kmin() == self.kmax();
147        match (i_same, j_same, k_same) {
148            (true, false, false) => Some(FaceAxis::I),
149            (false, true, false) => Some(FaceAxis::J),
150            (false, false, true) => Some(FaceAxis::K),
151            _ => None,
152        }

175    /// Length of the face diagonal between the extreme corner nodes.
176    pub fn diagonal_length(&self) -> f64 {
177        fn corner_index(face: &Face, imin: usize, jmin: usize, kmin: usize) -> usize {
178            face.indices
179                .iter()
180                .enumerate()
181                .find_map(|(idx, ijk)| {
182                    if ijk[0] == imin && ijk[1] == jmin && ijk[2] == kmin {
183                        Some(idx)
184                    } else {
185                        None
186                    }
187                })
188                .unwrap_or(0)
189        }
190
191        let min_idx = corner_index(self, self.imin(), self.jmin(), self.kmin());
192        let max_idx = corner_index(self, self.imax(), self.jmax(), self.kmax());
193        let p0 = self.vertices[min_idx];
194        let p1 = self.vertices[max_idx];
195        distance(p0, p1)
196    }

198    /// Compare vertex positions with a tolerance.
199    pub fn vertices_equals(&self, other: &Face, tol: f64) -> bool {
200        if self.vertices.len() != other.vertices.len() {
201            return false;
202        }
203        let mut matched = vec![false; other.vertices.len()];
204        for v in &self.vertices {
205            let mut found = false;
206            for (idx, o) in other.vertices.iter().enumerate() {
207                if matched[idx] {
208                    continue;
209                }
210                // deference and copy the values of v and o
211                if distance(*v, *o) <= tol {
212                    matched[idx] = true;
213                    found = true;
214                    break;
215                }
216            }
217            if !found {
218                return false;
219            }
220        }
221        true
222    }

224    /// Structured face points (dense sampling) for node matching.
225    ///
226    /// * `block` - Parent block.
227    /// * `stride_u`, `stride_v` - Sampling strides in parametric space.
228    pub fn grid_points(&self, block: &Block, stride_u: usize, stride_v: usize) -> Vec<[f64; 3]> {
229        let Some(axis) = self.const_axis() else {
230            return self.vertices.clone();
231        };
232        let su = stride_u.max(1);
233        let sv = stride_v.max(1);
234        let mut pts = Vec::new();
235        match axis {
236            FaceAxis::I => {
237                let i = self.imin();
238                for j in (self.jmin()..=self.jmax()).step_by(su) {
239                    for k in (self.kmin()..=self.kmax()).step_by(sv) {
240                        pts.push(to_array(block.xyz(i, j, k)));
241                    }
242                }
243            }
244            FaceAxis::J => {
245                let j = self.jmin();
246                for i in (self.imin()..=self.imax()).step_by(su) {
247                    for k in (self.kmin()..=self.kmax()).step_by(sv) {
248                        pts.push(to_array(block.xyz(i, j, k)));
249                    }
250                }
251            }
252            FaceAxis::K => {
253                let k = self.kmin();
254                for i in (self.imin()..=self.imax()).step_by(su) {
255                    for j in (self.jmin()..=self.jmax()).step_by(sv) {
256                        pts.push(to_array(block.xyz(i, j, k)));
257                    }
258                }
259            }
260        }
261        pts
262    }

264    /// Decide if another face shares enough nodes to be considered touching.
265    ///
266    /// * `other` - Candidate face.
267    /// * `block_self`, `block_other` - Parent blocks.
268    /// * `tol_xyz` - Distance tolerance for node equivalence.
269    /// * `min_shared_frac` - Minimum fraction of shared nodes.
270    /// * `min_shared_abs` - Minimum absolute number of shared nodes.
271    /// * `stride_u`, `stride_v` - Sampling stride along the face grid.
272    pub fn touches_by_nodes(
273        &self,
274        other: &Face,
275        block_self: &Block,
276        block_other: &Block,
277        tol_xyz: f64,
278        min_shared_frac: f64,
279        min_shared_abs: usize,
280        stride_u: usize,
281        stride_v: usize,
282    ) -> bool {
283        let pts_self = self.grid_points(block_self, stride_u, stride_v);
284        let pts_other = other.grid_points(block_other, stride_u, stride_v);
285        if pts_self.is_empty() || pts_other.is_empty() {
286            return false;
287        }
288
289        let q_self: HashSet<_> = pts_self
290            .iter()
291            .map(|p| quantize_point(*p, tol_xyz))
292            .collect();
293        let q_other: HashSet<_> = pts_other
294            .iter()
295            .map(|p| quantize_point(*p, tol_xyz))
296            .collect();
297
298        let shared = q_self.intersection(&q_other).count();
299        if shared < min_shared_abs {
300            return false;
301        }
302
303        let denom = pts_self.len().min(pts_other.len()) as f64;
304        (shared as f64) / denom >= min_shared_frac
305    }

346/// Helper structure representing a structured face grid.
347/// Dense representation of a structured face grid.
348#[derive(Clone, Debug)]
349pub struct StructuredFace {
350    /// Face dimensions `(nu, nv)`.
351    pub dims: (usize, usize),
352    /// Flattened coordinates stored row-major in `u`.
353    pub coords: Vec<[f64; 3]>,
354}

356impl StructuredFace {
357    fn idx(&self, u: usize, v: usize) -> [f64; 3] {
358        self.coords[v * self.dims.0 + u]
359    }

362#[derive(Copy, Clone, Debug)]
363enum BlockFaceKind {
364    IMin,
365    IMax,
366    JMin,
367    JMax,
368    KMin,
369    KMax,
370}
371
372impl BlockFaceKind {
373    fn all() -> [Self; 6] {
374        [
375            Self::IMin,
376            Self::IMax,
377            Self::JMin,
378            Self::JMax,
379            Self::KMin,
380            Self::KMax,
381        ]
382    }
383
384    fn name(self) -> &'static str {
385        match self {
386            Self::IMin => "imin",
387            Self::IMax => "imax",
388            Self::JMin => "jmin",
389            Self::JMax => "jmax",
390            Self::KMin => "kmin",
391            Self::KMax => "kmax",
392        }
393    }
394
395    fn dims(self, block: &Block) -> (usize, usize) {
396        match self {
397            Self::IMin | Self::IMax => (block.jmax, block.kmax),
398            Self::JMin | Self::JMax => (block.imax, block.kmax),
399            Self::KMin | Self::KMax => (block.imax, block.jmax),
400        }
401    }
402
403    fn sample(self, block: &Block, u: usize, v: usize) -> [f64; 3] {
404        match self {
405            Self::IMin => to_array(block.xyz(0, u, v)),
406            Self::IMax => to_array(block.xyz(block.imax - 1, u, v)),
407            Self::JMin => to_array(block.xyz(u, 0, v)),
408            Self::JMax => to_array(block.xyz(u, block.jmax - 1, v)),
409            Self::KMin => to_array(block.xyz(u, v, 0)),
410            Self::KMax => to_array(block.xyz(u, v, block.kmax - 1)),
411        }
412    }
413
414    fn structured_face(self, block: &Block) -> StructuredFace {
415        let dims = self.dims(block);
416        let mut coords = Vec::with_capacity(dims.0 * dims.1);
417        for v in 0..dims.1 {
418            for u in 0..dims.0 {
419                coords.push(self.sample(block, u, v));
420            }
421        }
422        StructuredFace { dims, coords }
423    }
424}

426/// Compute the Euclidean distance between two points.
427fn distance(a: [f64; 3], b: [f64; 3]) -> f64 {
428    let dx = a[0] - b[0];
429    let dy = a[1] - b[1];
430    let dz = a[2] - b[2];
431    (dx * dx + dy * dy + dz * dz).sqrt()
432}
433
434fn quantize_point(p: [f64; 3], tol: f64) -> (i64, i64, i64) {
435    let s = if tol > 0.0 { tol } else { DEFAULT_TOL };
436    (
437        (p[0] / s).round() as i64,
438        (p[1] / s).round() as i64,
439        (p[2] / s).round() as i64,
440    )
441}

448/// Deduplicate index pairs (order-agnostic).
449///
450/// # Arguments
451/// * `pairs` - Candidate index tuples `(a, b)`.
452///
453/// # Returns
454/// Deduplicated list preserving the original ordering of the input.
455pub fn unique_pairs(pairs: &[(usize, usize)]) -> Vec<(usize, usize)> {
456    let mut seen = HashSet::new();
457    let mut out = Vec::new();
458    for &(a, b) in pairs {
459        if a == b {
460            continue;
461        }
462        let key = if a < b { (a, b) } else { (b, a) };
463        if seen.insert(key) {
464            out.push((a, b));
465        }
466    }
467    out
468}

470/// Compare two structured faces and determine whether they match.
471///
472/// # Arguments
473/// * `face1` - Face sampled from the first block.
474/// * `face2` - Face sampled from the second block.
475/// * `tol` - Maximum Euclidean distance allowed between corner nodes.
476///
477/// # Returns
478/// `(matches, flips)` where `flips` encodes `(flip_ud, flip_lr)` applied to `face2`.
479pub fn faces_match(
480    face1: &StructuredFace,
481    face2: &StructuredFace,
482    tol: f64,
483) -> (bool, Option<(bool, bool)>) {
484    if face1.dims != face2.dims {
485        return (false, None);
486    }
487    let (ni, nj) = face1.dims;
488
489    let corners = |f: &StructuredFace, flip_ud: bool, flip_lr: bool| -> [[f64; 3]; 4] {
490        let map = |u: usize, v: usize| {
491            let uu = if flip_lr { ni - 1 - u } else { u };
492            let vv = if flip_ud { nj - 1 - v } else { v };
493            f.idx(uu, vv)
494        };
495        [
496            map(0, 0),
497            map(0, nj - 1),
498            map(ni - 1, 0),
499            map(ni - 1, nj - 1),
500        ]
501    };
502
503    let c1 = corners(face1, false, false);
504    for flip_ud in [false, true] {
505        for flip_lr in [false, true] {
506            let c2 = corners(face2, flip_ud, flip_lr);
507            if c1.iter().zip(&c2).all(|(a, b)| distance(*a, *b) <= tol) {
508                return (true, Some((flip_ud, flip_lr)));
509            }
510        }
511    }
512    (false, None)
513}

515/// Determine whether any faces on two blocks match.
516///
517/// # Arguments
518/// * `block1` - First block to compare.
519/// * `block2` - Second block to compare.
520/// * `tol` - Corner matching tolerance.
521///
522/// # Returns
523/// `Some((face_name_block1, face_name_block2, (flip_ud, flip_lr)))` when matching faces are found.
524pub fn find_matching_faces(
525    block1: &Block,
526    block2: &Block,
527    tol: f64,
528) -> Option<(&'static str, &'static str, (bool, bool))> {
529    for f1 in BlockFaceKind::all() {
530        let face1 = f1.structured_face(block1);
531        for f2 in BlockFaceKind::all() {
532            let face2 = f2.structured_face(block2);
533            let (matched, flips) = faces_match(&face1, &face2, tol);
534            if matched {
535                return flips.map(|flip| (f1.name(), f2.name(), flip));
536            }
537        }
538    }
539    None
540}

542/// Build the six outer faces for a block and identify internal matches.
543///
544/// # Arguments
545/// * `block` - Target plot3d block.
546///
547/// # Returns
548/// Tuple containing the exterior faces and any internal matching face pairs.
549pub fn get_outer_faces(block: &Block) -> (Vec<Face>, Vec<(Face, Face)>) {
550    let mut faces = Vec::with_capacity(6);
551    for kind in BlockFaceKind::all() {
552        let mut face = Face::new();
553        match kind {
554            BlockFaceKind::IMin | BlockFaceKind::IMax => {
555                let i = if matches!(kind, BlockFaceKind::IMin) {
556                    0
557                } else {
558                    block.imax - 1
559                };
560                for j in [0, block.jmax - 1] {
561                    for k in [0, block.kmax - 1] {
562                        let (x, y, z) = block.xyz(i, j, k);
563                        face.add_vertex(x, y, z, i, j, k);
564                    }
565                }
566            }
567            BlockFaceKind::JMin | BlockFaceKind::JMax => {
568                let j = if matches!(kind, BlockFaceKind::JMin) {
569                    0
570                } else {
571                    block.jmax - 1
572                };
573                for i in [0, block.imax - 1] {
574                    for k in [0, block.kmax - 1] {
575                        let (x, y, z) = block.xyz(i, j, k);
576                        face.add_vertex(x, y, z, i, j, k);
577                    }
578                }
579            }
580            BlockFaceKind::KMin | BlockFaceKind::KMax => {
581                let k = if matches!(kind, BlockFaceKind::KMin) {
582                    0
583                } else {
584                    block.kmax - 1
585                };
586                for i in [0, block.imax - 1] {
587                    for j in [0, block.jmax - 1] {
588                        let (x, y, z) = block.xyz(i, j, k);
589                        face.add_vertex(x, y, z, i, j, k);
590                    }
591                }
592            }
593        }
594        faces.push(face);
595    }
596
597    let mut matching_pairs = Vec::new();
598    let mut non_matching = Vec::new();
599    for i in 0..faces.len() {
600        let mut matched = false;
601        for j in 0..faces.len() {
602            if i == j {
603                continue;
604            }
605            if faces[i].vertices_equals(&faces[j], DEFAULT_TOL) {
606                matching_pairs.push((i, j));
607                matched = true;
608            }
609        }
610        if !matched {
611            non_matching.push(faces[i].clone());
612        }
613    }
614
615    let pairs = unique_pairs(&matching_pairs)
616        .into_iter()
617        .map(|(a, b)| (faces[a].clone(), faces[b].clone()))
618        .collect();
619
620    (non_matching, pairs)
621}

623/// Build a face from diagonal index pairs on a block.
624///
625/// # Arguments
626/// * `block` - Parent block.
627/// * `imin`, `jmin`, `kmin` - Lower corner indices.
628/// * `imax`, `jmax`, `kmax` - Upper corner indices.
629///
630/// # Returns
631/// New `Face` populated with the four corner nodes.
632pub fn create_face_from_diagonals(
633    block: &Block,
634    imin: usize,
635    jmin: usize,
636    kmin: usize,
637    imax: usize,
638    jmax: usize,
639    kmax: usize,
640) -> Face {
641    let mut face = Face::new();
642    if imin == imax {
643        let i = imin;
644        for j in [jmin, jmax] {
645            for k in [kmin, kmax] {
646                let (x, y, z) = block.xyz(i, j, k);
647                face.add_vertex(x, y, z, i, j, k);
648            }
649        }
650    } else if jmin == jmax {
651        let j = jmin;
652        for i in [imin, imax] {
653            for k in [kmin, kmax] {
654                let (x, y, z) = block.xyz(i, j, k);
655                face.add_vertex(x, y, z, i, j, k);
656            }
657        }
658    } else if kmin == kmax {
659        let k = kmin;
660        for i in [imin, imax] {
661            for j in [jmin, jmax] {
662                let (x, y, z) = block.xyz(i, j, k);
663                face.add_vertex(x, y, z, i, j, k);
664            }
665        }
666    }
667    face
668}

670/// Convert serialized face records back into `Face` instances.
671///
672/// # Arguments
673/// * `blocks` - Blocks interpreted at the reduced resolution.
674/// * `outer_faces` - Collection of serialized face records.
675/// * `gcd` - Grid reduction factor applied to the blocks.
676///
677/// # Returns
678/// Converted faces with block indices preserved.
679pub fn outer_face_records_to_list(
680    blocks: &[Block],
681    outer_faces: &[FaceRecord],
682    gcd: usize,
683) -> Vec<Face> {
684    let mut faces = Vec::new();
685    for record in outer_faces {
686        let block_idx = record.block_index;
687        if block_idx >= blocks.len() {
688            continue;
689        }
690        let block = &blocks[block_idx];
691        let scale = gcd.max(1);
692        let mut face = create_face_from_diagonals(
693            block,
694            record.imin / scale,
695            record.jmin / scale,
696            record.kmin / scale,
697            record.imax / scale,
698            record.jmax / scale,
699            record.kmax / scale,
700        );
701        face.set_block_index(block_idx);
702        if let Some(id) = record.id {
703            face.set_id(id);
704        }
705        faces.push(face);
706    }
707    faces
708}

748pub fn split_face(
749    face_to_split: &Face,

752    jmin: usize,
753    kmin: usize,
754    imax: usize,
755    jmax: usize,
756    kmax: usize,

759    let mut faces = Vec::new();
760
761    if kmin == kmax {
762        faces.push(create_face_from_diagonals(
763            block,
764            imin,
765            jmax,
766            kmin,
767            imax,
768            face_to_split.jmax(),
769            kmax,
770        ));
771        faces.push(create_face_from_diagonals(
772            block,
773            imin,
774            face_to_split.jmin(),
775            kmin,
776            imax,
777            jmin,
778            kmax,
779        ));

782            face_to_split.imin(),
783            face_to_split.jmin(),

785            imin,

787            kmax,
788        ));

790            block,
791            imax,
792            face_to_split.jmin(),
793            kmin,
794            face_to_split.imax(),
795            face_to_split.jmax(),
796            kmax,
797        ));
798    } else if imin == imax {
799        faces.push(create_face_from_diagonals(
800            block,
801            imin,
802            jmin,
803            kmax,
804            imax,
805            jmax,
806            face_to_split.kmax(),
807        ));
808        faces.push(create_face_from_diagonals(
809            block,
810            imin,
811            jmin,
812            face_to_split.kmin(),
813            imax,
814            jmax,
815            kmin,
816        ));
817        faces.push(create_face_from_diagonals(
818            block,
819            imin,
820            face_to_split.jmin(),
821            face_to_split.kmin(),
822            imax,
823            jmin,
824            face_to_split.kmax(),
825        ));
826        faces.push(create_face_from_diagonals(
827            block,
828            imin,
829            jmax,
830            face_to_split.kmin(),
831            imax,
832            face_to_split.jmax(),
833            face_to_split.kmax(),
834        ));
835    } else if jmin == jmax {
836        faces.push(create_face_from_diagonals(
837            block,
838            imin,
839            jmin,
840            kmax,
841            imax,
842            jmax,
843            face_to_split.kmax(),
844        ));
845        faces.push(create_face_from_diagonals(
846            block,
847            imin,
848            jmin,
849            face_to_split.kmin(),
850            imax,
851            jmax,
852            kmin,
853        ));
854        faces.push(create_face_from_diagonals(
855            block,
856            face_to_split.imin(),
857            jmin,
858            face_to_split.kmin(),
859            imin,
860            jmax,
861            face_to_split.kmax(),
862        ));
863        faces.push(create_face_from_diagonals(

879            } else {

882                }

884            }

889/// Pick the face closest to a reference point.
890///
891/// # Arguments
892/// * `faces` - Candidate faces.
893/// * `point` - Cartesian reference location.
894///
895/// # Returns
896/// Index of the nearest face or `None` when the list is empty.
897pub fn find_face_nearest_point(faces: &[Face], point: [f64; 3]) -> Option<usize> {
898    faces
899        .iter()
900        .enumerate()
901        .min_by(|(_, a), (_, b)| {
902            distance(a.centroid(), point)
903                .partial_cmp(&distance(b.centroid(), point))
904                .unwrap_or(std::cmp::Ordering::Equal)
905        })
906        .map(|(idx, _)| idx)
907}

909/// Reduce blocks by sampling every `factor` nodes along each axis.
910///
911/// # Arguments
912/// * `blocks` - Blocks to down-sample.
913/// * `factor` - Sampling step applied to i, j and k directions.
914///
915/// # Returns
916/// New blocks reduced to consistent spacing.
917pub fn reduce_blocks(blocks: &[Block], factor: usize) -> Vec<Block> {
918    if factor <= 1 {
919        return blocks.to_vec();
920    }
921    return blocks
922        .iter()
923        .map(|block| {
924            let si = (block.imax - 1) / factor + 1;
925            let sj = (block.jmax - 1) / factor + 1;
926            let sk = (block.kmax - 1) / factor + 1;
927            let mut x = Vec::with_capacity(si * sj * sk);
928            let mut y = Vec::with_capacity(si * sj * sk);
929            let mut z = Vec::with_capacity(si * sj * sk);
930            for k in (0..block.kmax).step_by(factor) {
931                for j in (0..block.jmax).step_by(factor) {
932                    for i in (0..block.imax).step_by(factor) {
933                        let (px, py, pz) = block.xyz(i, j, k);
934                        x.push(px);
935                        y.push(py);
936                        z.push(pz);
937                    }
938                }
939            }
940            Block::new(si, sj, sk, x, y, z)
941        })
942        .collect(); // Final expression is the return value
943}

945/// Rotate a block using a 3×3 rotation matrix.
946///
947/// # Arguments
948/// * `block` - Block to rotate.
949/// * `rotation` - Row-major rotation matrix.
950///
951/// # Returns
952/// Rotated block with identical dimensions.
953pub fn rotate_block(block: &Block, rotation: [[f64; 3]; 3]) -> Block {
954    let mut x = Vec::with_capacity(block.npoints());
955    let mut y = Vec::with_capacity(block.npoints());
956    let mut z = Vec::with_capacity(block.npoints());
957    for k in 0..block.kmax {
958        for j in 0..block.jmax {
959            for i in 0..block.imax {
960                let (px, py, pz) = block.xyz(i, j, k);
961                x.push(rotation[0][0] * px + rotation[0][1] * py + rotation[0][2] * pz);
962                y.push(rotation[1][0] * px + rotation[1][1] * py + rotation[1][2] * pz);
963                z.push(rotation[2][0] * px + rotation[2][1] * py + rotation[2][2] * pz);
964            }
965        }
966    }
967    return Block::new(block.imax, block.jmax, block.kmax, x, y, z);
968}

970/// Compute the global bounds across all blocks.
971///
972/// # Arguments
973/// * `blocks` - Collection of blocks to inspect.
974///
975/// # Returns
976/// `(x_bounds, y_bounds, z_bounds)` or `None` when the list is empty.
977pub fn get_outer_bounds(blocks: &[Block]) -> Option<((f64, f64), (f64, f64), (f64, f64))> {
978    if blocks.is_empty() {
979        return None;
980    }
981    let mut xmin = f64::INFINITY;
982    let mut xmax = f64::NEG_INFINITY;
983    let mut ymin = f64::INFINITY;
984    let mut ymax = f64::NEG_INFINITY;
985    let mut zmin = f64::INFINITY;
986    let mut zmax = f64::NEG_INFINITY;
987    for block in blocks {
988        for &val in block.x_slice() {
989            xmin = xmin.min(val);
990            xmax = xmax.max(val);
991        }
992        for &val in block.y_slice() {
993            ymin = ymin.min(val);
994            ymax = ymax.max(val);
995        }
996        for &val in block.z_slice() {
997            zmin = zmin.min(val);
998            zmax = zmax.max(val);
999        }
1000    }
1001    Some(((xmin, xmax), (ymin, ymax), (zmin, zmax)))
1002}

1030/// Connectivity matrices describing which faces touch between blocks.
1031///
1032/// # Arguments
1033/// * `blocks` - Original block list.
1034/// * `outer_faces` - Optional pre-computed outer faces (face records).
1035/// * `tol` - Compatibility parameter maintained for parity with Python (unused).
1036/// * `options` - Node matching thresholds and sampling strides.
1037///
1038/// # Returns
1039/// Four symmetric adjacency matrices for overall connectivity and each axis-specific match.
1040pub fn block_connection_matrix(
1041    blocks: &[Block],
1042    outer_faces: &[FaceRecord],
1043    tol: f64,
1044    options: BlockConnectionOptions,
1045) -> (Vec<Vec<i8>>, Vec<Vec<i8>>, Vec<Vec<i8>>, Vec<Vec<i8>>) {
1046    let gcd = blocks
1047        .iter()
1048        .map(|b| {
1049            gcd_three(
1050                b.imax.saturating_sub(1),
1051                b.jmax.saturating_sub(1),
1052                b.kmax.saturating_sub(1),
1053            )
1054        })
1055        .filter(|&g| g > 0)
1056        .min()
1057        .unwrap_or(1);
1058    let reduced = reduce_blocks(blocks, gcd);
1059
1060    let mut faces_by_block: Vec<Vec<Face>> = vec![Vec::new(); blocks.len()];
1061    if outer_faces.is_empty() {
1062        for (idx, block) in reduced.iter().enumerate() {
1063            let (faces, _) = get_outer_faces(block);
1064            faces_by_block[idx] = faces
1065                .into_iter()
1066                .map(|mut f| {
1067                    f.set_block_index(idx);
1068                    f
1069                })
1070                .collect();
1071        }
1072    } else {
1073        for face in outer_face_records_to_list(&reduced, outer_faces, gcd) {
1074            if let Some(idx) = face.block_index() {
1075                if idx < faces_by_block.len() {
1076                    faces_by_block[idx].push(face);
1077                }
1078            }
1079        }
1080    }
1081
1082    let n = blocks.len();
1083    let mut connectivity = vec![vec![0i8; n]; n];
1084    let mut conn_i = vec![vec![0i8; n]; n];
1085    let mut conn_j = vec![vec![0i8; n]; n];
1086    let mut conn_k = vec![vec![0i8; n]; n];
1087    for i in 0..n {
1088        connectivity[i][i] = 1;
1089        conn_i[i][i] = 1;
1090        conn_j[i][i] = 1;
1091        conn_k[i][i] = 1;
1092    }
1093
1094    for i in 0..n {
1095        for j in (i + 1)..n {
1096            let mut connected = false;
1097            for face_i in &faces_by_block[i] {
1098                for face_j in &faces_by_block[j] {
1099                    if face_i.touches_by_nodes(
1100                        face_j,
1101                        &reduced[i],
1102                        &reduced[j],
1103                        options.node_tol_xyz,
1104                        options.min_shared_frac,
1105                        options.min_shared_abs,
1106                        options.stride_u,
1107                        options.stride_v,
1108                    ) {
1109                        connectivity[i][j] = 1;
1110                        connectivity[j][i] = 1;
1111                        if face_i.const_axis() == Some(FaceAxis::I)
1112                            && face_j.const_axis() == Some(FaceAxis::I)
1113                        {
1114                            conn_i[i][j] = 1;
1115                            conn_i[j][i] = 1;
1116                        }
1117                        if face_i.const_axis() == Some(FaceAxis::J)
1118                            && face_j.const_axis() == Some(FaceAxis::J)
1119                        {
1120                            conn_j[i][j] = 1;
1121                            conn_j[j][i] = 1;
1122                        }
1123                        if face_i.const_axis() == Some(FaceAxis::K)
1124                            && face_j.const_axis() == Some(FaceAxis::K)
1125                        {
1126                            conn_k[i][j] = 1;
1127                            conn_k[j][i] = 1;
1128                        }
1129                        connected = true;
1130                        break;
1131                    }
1132                }
1133                if connected {
1134                    break;
1135                }
1136            }
1137            if !connected {
1138                connectivity[i][j] = -1;
1139                connectivity[j][i] = -1;
1140            }
1141        }
1142    }
1143
1144    if tol.is_finite() {
1145        let _ = tol; // placeholder for parity with Python signature
1146    }
1147
1148    (connectivity, conn_i, conn_j, conn_k)
1149}

1151fn gcd_two(mut a: usize, mut b: usize) -> usize {
1152    while b != 0 {
1153        let r = a % b;
1154        a = b;
1155        b = r;
1156    }
1157    a
1158}

1160fn gcd_three(a: usize, b: usize, c: usize) -> usize {
1161    gcd_two(gcd_two(a, b), c)
1162}

1164/// Standardise block orientation so that indices increase with coordinate values.
1165///
1166/// # Arguments
1167/// * `block` - Block whose axes may require flipping.
1168///
1169/// # Returns
1170/// New block with consistent orientation.
1171pub fn standardize_block_orientation(block: &Block) -> Block {
1172    let mut x = block.x.clone();
1173    let mut y = block.y.clone();
1174    let mut z = block.z.clone();
1175    let dims = (block.imax, block.jmax, block.kmax);
1176
1177    let center_i = block.imax / 2;
1178    let center_j = block.jmax / 2;
1179    let center_k = block.kmax / 2;
1180
1181    if block.imax > 1 {
1182        let delta =
1183            block.x_at(block.imax - 1, center_j, center_k) - block.x_at(0, center_j, center_k);
1184        if delta < 0.0 {
1185            flip_block_axis(&mut x, &mut y, &mut z, dims, 0);
1186        }
1187    }
1188    if block.jmax > 1 {
1189        let delta =
1190            block.y_at(center_i, block.jmax - 1, center_k) - block.y_at(center_i, 0, center_k);
1191        if delta < 0.0 {
1192            flip_block_axis(&mut x, &mut y, &mut z, dims, 1);
1193        }
1194    }
1195    if block.kmax > 1 {
1196        let delta =
1197            block.z_at(center_i, center_j, block.kmax - 1) - block.z_at(center_i, center_j, 0);
1198        if delta < 0.0 {
1199            flip_block_axis(&mut x, &mut y, &mut z, dims, 2);
1200        }
1201    }
1202
1203    Block::new(block.imax, block.jmax, block.kmax, x, y, z)
1204}

1206fn flip_block_axis(
1207    x: &mut [f64],
1208    y: &mut [f64],
1209    z: &mut [f64],
1210    dims: (usize, usize, usize),
1211    axis: usize,
1212) {
1213    let (imax, jmax, kmax) = dims;
1214    match axis {
1215        0 => {
1216            for k in 0..kmax {
1217                for j in 0..jmax {
1218                    for i in 0..imax / 2 {
1219                        let idx1 = (k * jmax + j) * imax + i;
1220                        let idx2 = (k * jmax + j) * imax + (imax - 1 - i);
1221                        x.swap(idx1, idx2);
1222                        y.swap(idx1, idx2);
1223                        z.swap(idx1, idx2);
1224                    }
1225                }
1226            }
1227        }
1228        1 => {
1229            for k in 0..kmax {
1230                for j in 0..jmax / 2 {
1231                    for i in 0..imax {
1232                        let idx1 = (k * jmax + j) * imax + i;
1233                        let idx2 = (k * jmax + (jmax - 1 - j)) * imax + i;
1234                        x.swap(idx1, idx2);
1235                        y.swap(idx1, idx2);
1236                        z.swap(idx1, idx2);
1237                    }
1238                }
1239            }
1240        }
1241        2 => {
1242            for k in 0..kmax / 2 {
1243                for j in 0..jmax {
1244                    for i in 0..imax {
1245                        let idx1 = (k * jmax + j) * imax + i;
1246                        let idx2 = ((kmax - 1 - k) * jmax + j) * imax + i;
1247                        x.swap(idx1, idx2);
1248                        y.swap(idx1, idx2);
1249                        z.swap(idx1, idx2);
1250                    }
1251                }
1252            }
1253        }
1254        _ => {}
1255    }
1256}

1258/// Simple collinearity test using the cross product.
1259///
1260/// # Arguments
1261/// * `v1` - First vector.
1262/// * `v2` - Second vector.
1263///
1264/// # Returns
1265/// `true` when the vectors are collinear.
1266pub fn check_collinearity(v1: [f64; 3], v2: [f64; 3]) -> bool {
1267    let cross = [
1268        v1[1] * v2[2] - v1[2] * v2[1],
1269        v1[2] * v2[0] - v1[0] * v2[2],
1270        v1[0] * v2[1] - v1[1] * v2[0],
1271    ];
1272    cross.iter().all(|c| c.abs() <= f64::EPSILON)
1273}

1275/// Compute outward normals for the six faces of a block.
1276///
1277/// # Arguments
1278/// * `block` - Block whose outward normals are required.
1279///
1280/// # Returns
1281/// Tuple containing normals for `(Imin, Jmin, Kmin, Imax, Jmax, Kmax)`.
1282pub fn calculate_outward_normals(
1283    block: &Block,
1284) -> ([f64; 3], [f64; 3], [f64; 3], [f64; 3], [f64; 3], [f64; 3]) {
1285    let i0 = to_array(block.xyz(0, 0, 0));
1286    let ij = to_array(block.xyz(0, block.jmax - 1, 0));
1287    let ik = to_array(block.xyz(0, 0, block.kmax - 1));
1288    let ni_min = cross(sub(ij, i0), sub(ik, i0));
1289
1290    let i1 = to_array(block.xyz(block.imax - 1, 0, 0));
1291    let i1j = to_array(block.xyz(block.imax - 1, block.jmax - 1, 0));
1292    let i1k = to_array(block.xyz(block.imax - 1, 0, block.kmax - 1));
1293    let ni_max = cross(sub(i1j, i1), sub(i1k, i1));
1294
1295    let j0 = to_array(block.xyz(0, 0, 0));
1296    let ji = to_array(block.xyz(block.imax - 1, 0, 0));
1297    let jk = to_array(block.xyz(0, 0, block.kmax - 1));
1298    let nj_min = cross(sub(ji, j0), sub(jk, j0));
1299
1300    let j1 = to_array(block.xyz(0, block.jmax - 1, 0));
1301    let j1i = to_array(block.xyz(block.imax - 1, block.jmax - 1, 0));
1302    let j1k = to_array(block.xyz(0, block.jmax - 1, block.kmax - 1));
1303    let nj_max = cross(sub(j1i, j1), sub(j1k, j1));
1304
1305    let k0 = to_array(block.xyz(0, 0, 0));
1306    let ki = to_array(block.xyz(block.imax - 1, 0, 0));
1307    let kj = to_array(block.xyz(0, block.jmax - 1, 0));
1308    let nk_min = cross(sub(ki, k0), sub(kj, k0));
1309
1310    let k1 = to_array(block.xyz(0, 0, block.kmax - 1));
1311    let k1i = to_array(block.xyz(block.imax - 1, 0, block.kmax - 1));
1312    let k1j = to_array(block.xyz(0, block.jmax - 1, block.kmax - 1));
1313    let nk_max = cross(sub(k1i, k1), sub(k1j, k1));
1314
1315    (ni_min, nj_min, nk_min, ni_max, nj_max, nk_max)
1316}

1322fn cross(a: [f64; 3], b: [f64; 3]) -> [f64; 3] {
1323    [
1324        a[1] * b[2] - a[2] * b[1],
1325        a[2] * b[0] - a[0] * b[2],
1326        a[0] * b[1] - a[1] * b[0],
1327    ]
1328}

1330/// Identify outer faces on the extreme of the requested axis using BFS.
1331///
1332/// # Arguments
1333/// * `blocks` - All blocks in the system.
1334/// * `outer_faces_dicts` - Optional pre-computed outer faces in dictionary form.
1335/// * `direction` - Axis name (`"x"`, `"y"`, or `"z"`).
1336/// * `side` - Requested side (`"min"`, `"max"`, or `"both"`).
1337/// * `tol_rel` - Relative tolerance for plane selection.
1338/// * `node_tol_xyz` - Node matching tolerance for BFS linking.
1339///
1340/// # Returns
1341/// Tuple containing serialized faces and raw face objects for the lower and upper planes.
1342pub fn find_bounding_faces(
1343    blocks: &[Block],
1344    outer_faces_records: &[FaceRecord],
1345    direction: &str,
1346    side: &str,
1347    tol_rel: f64,
1348    node_tol_xyz: f64,
1349) -> (Vec<FaceRecord>, Vec<FaceRecord>, Vec<Face>, Vec<Face>) {
1350    if blocks.is_empty() {
1351        return (Vec::new(), Vec::new(), Vec::new(), Vec::new());
1352    }
1353    let axis = match direction {
1354        "x" => FaceAxis::I,
1355        "y" => FaceAxis::J,
1356        _ => FaceAxis::K,
1357    };
1358    let want_min = side == "min" || side == "both";
1359    let want_max = side == "max" || side == "both";
1360
1361    let gcd = blocks
1362        .iter()
1363        .map(|b| gcd_three(b.imax - 1, b.jmax - 1, b.kmax - 1))
1364        .filter(|&g| g > 0)
1365        .min()
1366        .unwrap_or(1);
1367    let reduced = reduce_blocks(blocks, gcd);
1368
1369    let outer_faces = if outer_faces_records.is_empty() {
1370        reduced
1371            .iter()
1372            .enumerate()
1373            .flat_map(|(idx, block)| {
1374                let (faces, _) = get_outer_faces(block);
1375                faces.into_iter().map(move |mut f| {
1376                    f.set_block_index(idx);
1377                    f
1378                })
1379            })
1380            .collect::<Vec<_>>()
1381    } else {
1382        outer_face_records_to_list(&reduced, outer_faces_records, gcd)
1383    };
1384
1385    let axis_range = global_axis_bounds(&reduced, axis).unwrap_or((0.0, 0.0));
1386    let tol_abs = tol_rel * (axis_range.0.abs() + axis_range.1.abs()).max(1.0);
1387
1388    let mut lower = Vec::new();
1389    let mut upper = Vec::new();
1390
1391    if want_min {
1392        lower = collect_boundary_faces(&outer_faces, &reduced, axis, true, tol_abs, node_tol_xyz);
1393    }
1394    if want_max {
1395        upper = collect_boundary_faces(&outer_faces, &reduced, axis, false, tol_abs, node_tol_xyz);
1396    }
1397
1398    let lower_export = lower.iter().map(Face::to_record).collect();
1399    let upper_export = upper.iter().map(Face::to_record).collect();
1400    (lower_export, upper_export, lower, upper)
1401}

1403fn global_axis_bounds(blocks: &[Block], axis: FaceAxis) -> Option<(f64, f64)> {
1404    let mut min_val = f64::INFINITY;
1405    let mut max_val = f64::NEG_INFINITY;
1406    for block in blocks {
1407        match axis {
1408            FaceAxis::I => {
1409                for &x in block.x_slice() {
1410                    min_val = min_val.min(x);
1411                    max_val = max_val.max(x);
1412                }
1413            }
1414            FaceAxis::J => {
1415                for &y in block.y_slice() {
1416                    min_val = min_val.min(y);
1417                    max_val = max_val.max(y);
1418                }
1419            }
1420            FaceAxis::K => {
1421                for &z in block.z_slice() {
1422                    min_val = min_val.min(z);
1423                    max_val = max_val.max(z);
1424                }
1425            }
1426        }
1427    }
1428    if min_val.is_finite() && max_val.is_finite() {
1429        Some((min_val, max_val))
1430    } else {
1431        None
1432    }
1433}

1435fn collect_boundary_faces(
1436    faces: &[Face],
1437    blocks: &[Block],
1438    axis: FaceAxis,
1439    is_min: bool,
1440    tol_abs: f64,
1441    node_tol_xyz: f64,
1442) -> Vec<Face> {
1443    if faces.is_empty() {
1444        return Vec::new();
1445    }
1446
1447    let mut plane_value = if is_min {
1448        f64::INFINITY
1449    } else {
1450        f64::NEG_INFINITY
1451    };
1452    for face in faces {
1453        for v in &face.vertices {
1454            let val = match axis {
1455                FaceAxis::I => v[0],
1456                FaceAxis::J => v[1],
1457                FaceAxis::K => v[2],
1458            };
1459            if is_min {
1460                plane_value = plane_value.min(val);
1461            } else {
1462                plane_value = plane_value.max(val);
1463            }
1464        }
1465    }
1466    if !plane_value.is_finite() {
1467        return Vec::new();
1468    }
1469
1470    let mut plane_faces = Vec::new();
1471    for face in faces {
1472        let mut fmin = f64::INFINITY;
1473        let mut fmax = f64::NEG_INFINITY;
1474        for v in &face.vertices {
1475            let val = match axis {
1476                FaceAxis::I => v[0],
1477                FaceAxis::J => v[1],
1478                FaceAxis::K => v[2],
1479            };
1480            fmin = fmin.min(val);
1481            fmax = fmax.max(val);
1482        }
1483        let touches_plane = if is_min {
1484            (fmin - plane_value).abs() <= tol_abs
1485        } else {
1486            (fmax - plane_value).abs() <= tol_abs
1487        };
1488        let not_past = if is_min {
1489            (fmax - plane_value) <= tol_abs
1490        } else {
1491            (plane_value - fmin) <= tol_abs
1492        };
1493        if touches_plane && not_past {
1494            plane_faces.push(face.clone());
1495        }
1496    }
1497
1498    let mut visited: HashSet<(usize, usize, usize, usize, usize, usize, usize)> = HashSet::new();
1499    let mut result = Vec::new();
1500    for seed in &plane_faces {
1501        let mut queue = VecDeque::new();
1502        queue.push_back(seed.clone());
1503        while let Some(face) = queue.pop_front() {
1504            let key = face.index_key();
1505            if !visited.insert(key) {
1506                continue;
1507            }
1508            result.push(face.clone());
1509            for candidate in &plane_faces {
1510                let cand_key = candidate.index_key();
1511                if visited.contains(&cand_key) {
1512                    continue;
1513                }
1514                let Some(a_idx) = face.block_index() else {
1515                    continue;
1516                };
1517                let Some(b_idx) = candidate.block_index() else {
1518                    continue;
1519                };
1520                if a_idx >= blocks.len() || b_idx >= blocks.len() {
1521                    continue;
1522                }
1523                if face.touches_by_nodes(
1524                    candidate,
1525                    &blocks[a_idx],
1526                    &blocks[b_idx],
1527                    node_tol_xyz,
1528                    0.02,
1529                    2,
1530                    1,
1531                    1,
1532                ) {
1533                    queue.push_back(candidate.clone());
1534                }
1535            }
1536        }
1537    }
1538    result
1539}

1541/// Find the block whose centroid is closest to an extrapolated target.
1542///
1543/// # Arguments
1544/// * `blocks` - Candidate blocks.
1545/// * `centroid` - Reference centroid for the entire assembly.
1546/// * `direction` - Axis name controlling the search direction.
1547/// * `minvalue` - When `true`, search toward the minimum extreme; otherwise the maximum.
1548///
1549/// # Returns
1550/// The selected block index and the target coordinates used for the comparison.
1551pub fn find_closest_block(
1552    blocks: &[Block],
1553    centroid: [f64; 3],
1554    direction: &str,
1555    minvalue: bool,
1556) -> Option<(usize, f64, f64, f64)> {
1557    let Some((xbounds, ybounds, zbounds)) = get_outer_bounds(blocks) else {
1558        return None;
1559    };
1560    let (target_x, target_y, target_z) = match direction {
1561        "x" => {
1562            let dx = xbounds.1 - xbounds.0;
1563            let x = if minvalue {
1564                xbounds.0 - 0.5 * dx
1565            } else {
1566                xbounds.1 + 0.5 * dx
1567            };
1568            (x, centroid[1], centroid[2])
1569        }
1570        "y" => {
1571            let dy = ybounds.1 - ybounds.0;
1572            let y = if minvalue {
1573                ybounds.0 - 0.5 * dy
1574            } else {
1575                ybounds.1 + 0.5 * dy
1576            };
1577            (centroid[0], y, centroid[2])
1578        }
1579        _ => {
1580            let dz = zbounds.1 - zbounds.0;
1581            let z = if minvalue {
1582                zbounds.0 - 0.5 * dz
1583            } else {
1584                zbounds.1 + 0.5 * dz
1585            };
1586            (centroid[0], centroid[1], z)
1587        }
1588    };
1589    let mut best_idx = None;
1590    let mut best_dist = f64::INFINITY;
1591    for (idx, block) in blocks.iter().enumerate() {
1592        let cx = block.x_slice().iter().sum::<f64>() / block.x_slice().len() as f64;
1593        let cy = block.y_slice().iter().sum::<f64>() / block.y_slice().len() as f64;
1594        let cz = block.z_slice().iter().sum::<f64>() / block.z_slice().len() as f64;
1595        let dist = distance([cx, cy, cz], [target_x, target_y, target_z]);
1596        if dist < best_dist {
1597            best_dist = dist;
1598            best_idx = Some(idx);
1599        }
1600    }
1601    best_idx.map(|idx| (idx, target_x, target_y, target_z))
1602}

1604/// Graph helper: find a neighbour connected to both `a` and `b`.
1605///
1606/// # Arguments
1607/// * `graph` - Adjacency map.
1608/// * `a` - First node.
1609/// * `b` - Second node.
1610/// * `exclude` - Nodes that must not be returned.
1611///
1612/// # Returns
1613/// `Some(node)` when a mutual neighbour exists.
1614pub fn common_neighbor(
1615    graph: &HashMap<usize, HashSet<usize>>,
1616    a: usize,
1617    b: usize,
1618    exclude: &HashSet<usize>,
1619) -> Option<usize> {
1620    graph
1621        .get(&a)?
1622        .iter()
1623        .find(|&&n| {
1624            n != b && !exclude.contains(&n) && graph.get(&n).map_or(false, |s| s.contains(&b))
1625        })
1626        .copied()
1627}

1629/// Convert face matches into an adjacency map.
1630///
1631/// # Arguments
1632/// * `connectivities` - Matched face descriptions (block1/block2 pairings).
1633///
1634/// # Returns
1635/// Undirected adjacency map between block indices.
1636pub fn build_connectivity_graph(connectivities: &[FaceMatch]) -> HashMap<usize, HashSet<usize>> {
1637    let mut graph: HashMap<usize, HashSet<usize>> = HashMap::new();
1638    for pair in connectivities {
1639        let block1 = pair.block1.block_index;
1640        let block2 = pair.block2.block_index;
1641        if block1 == usize::MAX || block2 == usize::MAX {
1642            continue;
1643        }
1644        graph.entry(block1).or_default().insert(block2);
1645        graph.entry(block2).or_default().insert(block1);
1646    }
1647    graph

17/// Detect translational periodicity along an axis.
18/// Discover translational periodicity along a chosen axis.
19///
20/// # Testing
21/// End-to-end validation is planned to follow the pattern established in
22/// `tests/test_rotational_periodicity.rs`. Until then, exercising this function in a binary or
23/// ad-hoc script is recommended to mirror the original Python examples.
24pub fn translational_periodicity(
25    blocks: &[Block],
26    outer_faces: &[FaceRecord],
27    delta: Option<f64>,
28    translational_direction: &str,
29    node_tol_xyz: Option<f64>,
30    min_shared_frac: f64,
31    min_shared_abs: usize,
32    stride_u: usize,
33    stride_v: usize,
34) -> (Vec<FaceMatch>, Vec<FaceRecord>) {
35    if blocks.is_empty() {
36        return (Vec::new(), Vec::new());
37    }
38
39    let axis = translational_direction.trim().to_ascii_lowercase();
40    assert!(matches!(axis.as_str(), "x" | "y" | "z"));
41
42    let (lower_faces_records, upper_faces_records, _, _) =
43        find_bounding_faces(blocks, outer_faces, &axis, "both", 1e-6, 1e-6);
44
45    let mut gcd_array = Vec::new();
46    for block in blocks {
47        gcd_array.push(gcd_three(block.imax - 1, block.jmax - 1, block.kmax - 1));
48    }
49    let gcd_to_use = gcd_array.into_iter().min().unwrap_or(1).max(1);
50
51    let blocks_reduced = crate::block_face_functions::reduce_blocks(blocks, gcd_to_use);
52    let lower_faces = outer_face_records_to_list(&blocks_reduced, &lower_faces_records, 1);
53    let upper_faces = outer_face_records_to_list(&blocks_reduced, &upper_faces_records, 1);
54
55    let delta_axis = match axis.as_str() {
56        "x" => {
57            let min_x = blocks_reduced
58                .iter()
59                .map(|b| b.x_slice().iter().cloned().fold(f64::INFINITY, f64::min))
60                .fold(f64::INFINITY, f64::min);
61            let max_x = blocks_reduced
62                .iter()
63                .map(|b| {
64                    b.x_slice()
65                        .iter()
66                        .cloned()
67                        .fold(f64::NEG_INFINITY, f64::max)
68                })
69                .fold(f64::NEG_INFINITY, f64::max);
70            delta.unwrap_or(max_x - min_x)
71        }
72        "y" => {
73            let min_y = blocks_reduced
74                .iter()
75                .map(|b| b.y_slice().iter().cloned().fold(f64::INFINITY, f64::min))
76                .fold(f64::INFINITY, f64::min);
77            let max_y = blocks_reduced
78                .iter()
79                .map(|b| {
80                    b.y_slice()
81                        .iter()
82                        .cloned()
83                        .fold(f64::NEG_INFINITY, f64::max)
84                })
85                .fold(f64::NEG_INFINITY, f64::max);
86            delta.unwrap_or(max_y - min_y)
87        }
88        _ => {
89            let min_z = blocks_reduced
90                .iter()
91                .map(|b| b.z_slice().iter().cloned().fold(f64::INFINITY, f64::min))
92                .fold(f64::INFINITY, f64::min);
93            let max_z = blocks_reduced
94                .iter()
95                .map(|b| {
96                    b.z_slice()
97                        .iter()
98                        .cloned()
99                        .fold(f64::NEG_INFINITY, f64::max)
100                })
101                .fold(f64::NEG_INFINITY, f64::max);
102            delta.unwrap_or(max_z - min_z)
103        }
104    };
105
106    let blocks_up: Vec<Block> = blocks_reduced
107        .iter()
108        .map(|b| b.shifted(delta_axis, axis.chars().next().unwrap()))
109        .collect();
110    let blocks_dn: Vec<Block> = blocks_reduced
111        .iter()
112        .map(|b| b.shifted(-delta_axis, axis.chars().next().unwrap()))
113        .collect();
114
115    let mut periodic_matches = Vec::new();
116
117    let lower_pool = dedup_faces(lower_faces);
118    let mut upper_pool = dedup_faces(upper_faces);
119
120    for face_l in lower_pool.clone() {
121        let candidate = upper_pool.iter().enumerate().find_map(|(idx, f)| {
122            faces_translational_match(
123                &face_l,
124                f,
125                &blocks_reduced,
126                &blocks_up,
127                &blocks_dn,
128                axis.as_str(),
129                delta_axis,
130                node_tol_xyz,
131                min_shared_frac,
132                min_shared_abs,
133                stride_u,
134                stride_v,
135            )
136            .map(|mode| (idx, mode))
137        });
138        if let Some((pos, _mode)) = candidate {
139            let face_u = upper_pool.remove(pos);
140            periodic_matches.push(FaceMatch {
141                block1: FaceRecord::from_face(&face_l),
142                block2: FaceRecord::from_face(&face_u),
143                points: Vec::new(),
144            });
145        }
146    }
147
148    let mut periodic_keys = HashSet::new();
149    for rec in &periodic_matches {
150        periodic_keys.insert(face_record_key(&rec.block1));
151        periodic_keys.insert(face_record_key(&rec.block2));
152    }
153
154    let mut remaining = Vec::new();
155    for record in outer_faces {
156        if !periodic_keys.contains(&face_record_key(record)) {
157            remaining.push(record.clone());
158        }
159    }
160
161    if gcd_to_use > 1 {
162        for rec in &mut periodic_matches {
163            rec.block1.scale_indices(gcd_to_use);
164            rec.block2.scale_indices(gcd_to_use);
165        }
166        for record in &mut remaining {
167            record.scale_indices(gcd_to_use);
168        }
169    }
170
171    (periodic_matches, remaining)
172}

174/// Assess one lower/upper face combo and return the match mode when successful.
175fn faces_translational_match(
176    face_l: &Face,
177    face_u: &Face,
178    blocks: &[Block],
179    blocks_up: &[Block],
180    blocks_dn: &[Block],
181    axis: &str,
182    delta_axis: f64,
183    node_tol_xyz: Option<f64>,
184    min_shared_frac: f64,
185    min_shared_abs: usize,
186    stride_u: usize,
187    stride_v: usize,
188) -> Option<String> {
189    let tol_pair = pair_tolerance(face_l, face_u, blocks, node_tol_xyz, axis);
190
191    if orthogonal_precheck(
192        face_l,
193        face_u,
194        &blocks_up[face_l.block_index().unwrap()],
195        &blocks[face_u.block_index().unwrap()],
196        delta_axis,
197        tol_pair,
198        axis,
199        min_shared_frac,
200        min_shared_abs,
201    ) {
202        return Some(format!("{axis}_precheck_lower_up"));
203    }
204    if face_l.touches_by_nodes(
205        face_u,
206        &blocks_up[face_l.block_index().unwrap()],
207        &blocks[face_u.block_index().unwrap()],
208        tol_pair,
209        min_shared_frac,
210        min_shared_abs,
211        stride_u,
212        stride_v,
213    ) {
214        return Some("lower_up_vs_upper_orig".to_string());
215    }
216    if face_l.touches_by_nodes(
217        face_u,
218        &blocks[face_l.block_index().unwrap()],
219        &blocks_dn[face_u.block_index().unwrap()],
220        tol_pair,
221        min_shared_frac,
222        min_shared_abs,
223        stride_u,
224        stride_v,
225    ) {
226        return Some("lower_orig_vs_upper_dn".to_string());
227    }
228    if orthogonal_precheck(
229        face_u,
230        face_l,
231        &blocks_up[face_u.block_index().unwrap()],
232        &blocks[face_l.block_index().unwrap()],
233        delta_axis,
234        tol_pair,
235        axis,
236        min_shared_frac,
237        min_shared_abs,
238    ) {
239        return Some(format!("{axis}_precheck_upper_up"));
240    }
241    if face_u.touches_by_nodes(
242        face_l,
243        &blocks_up[face_u.block_index().unwrap()],
244        &blocks[face_l.block_index().unwrap()],
245        tol_pair,
246        min_shared_frac,
247        min_shared_abs,
248        stride_u,
249        stride_v,
250    ) {
251        return Some("upper_up_vs_lower_orig".to_string());
252    }
253    face_u
254        .touches_by_nodes(
255            face_l,
256            &blocks[face_u.block_index().unwrap()],
257            &blocks_dn[face_l.block_index().unwrap()],
258            tol_pair,
259            min_shared_frac,
260            min_shared_abs,
261            stride_u,
262            stride_v,
263        )
264        .then(|| "upper_orig_vs_lower_dn".to_string())
265}

267/// Decide the XYZ tolerance for a particular face pair, optionally honoring a global override.
268fn pair_tolerance(
269    face_a: &Face,
270    face_b: &Face,
271    blocks: &[Block],
272    override_tol: Option<f64>,
273    axis: &str,
274) -> f64 {
275    if let Some(tol) = override_tol {
276        return tol;
277    }
278    let spacing_a = median_inplane_spacing(face_a, &blocks[face_a.block_index().unwrap()], axis);
279    let spacing_b = median_inplane_spacing(face_b, &blocks[face_b.block_index().unwrap()], axis);
280    (0.03 * spacing_a.max(spacing_b)).max(1e-4)
281}

283/// Compute a median edge length for the face in the non-periodic directions.
284fn median_inplane_spacing(face: &Face, block: &Block, axis: &str) -> f64 {
285    let points = face.grid_points(block, 1, 1);
286    if points.len() <= 1 {
287        return 1.0;
288    }
289    let mut spacings = Vec::new();
290    for window in points.windows(2) {
291        let p0 = window[0];
292        let p1 = window[1];
293        let diff = match axis {
294            "x" => [(p0[1] - p1[1]).abs(), (p0[2] - p1[2]).abs()],
295            "y" => [(p0[0] - p1[0]).abs(), (p0[2] - p1[2]).abs()],
296            _ => [(p0[0] - p1[0]).abs(), (p0[1] - p1[1]).abs()],
297        };
298        spacings.push(diff[0].hypot(diff[1]));
299    }
300    spacings.sort_by(|a, b| a.partial_cmp(b).unwrap());
301    spacings[spacings.len() / 2]
302}

304/// Perform a quick planar projection test to reject clearly mismatched faces.
305fn orthogonal_precheck(
306    face_a: &Face,
307    face_b: &Face,
308    block_a: &Block,
309    block_b: &Block,
310    delta: f64,
311    tol: f64,
312    axis: &str,
313    min_shared_frac: f64,
314    min_shared_abs: usize,
315) -> bool {
316    let mut pts_a = face_a.grid_points(block_a, 1, 1);
317    let pts_b = face_b.grid_points(block_b, 1, 1);
318    if pts_a.is_empty() || pts_b.is_empty() {
319        return false;
320    }
321    match axis {
322        "x" => pts_a.iter_mut().for_each(|p| p[0] += delta),
323        "y" => pts_a.iter_mut().for_each(|p| p[1] += delta),
324        _ => pts_a.iter_mut().for_each(|p| p[2] += delta),
325    }
326
327    let proj_a = project_plane(&pts_a, axis);
328    let proj_b = project_plane(&pts_b, axis);
329
330    let key_a: HashSet<(i64, i64)> = proj_a
331        .iter()
332        .map(|p| ((p[0] / tol).round() as i64, (p[1] / tol).round() as i64))
333        .collect();
334    let key_b: HashSet<(i64, i64)> = proj_b
335        .iter()
336        .map(|p| ((p[0] / tol).round() as i64, (p[1] / tol).round() as i64))
337        .collect();
338
339    let shared = key_a.intersection(&key_b).count();
340    shared >= min_shared_abs
341        && (shared as f64) >= min_shared_frac * (key_a.len().min(key_b.len()) as f64)
342}

344/// Project 3D points onto the plane orthogonal to `axis`.
345fn project_plane(points: &[[f64; 3]], axis: &str) -> Vec<[f64; 2]> {
346    points
347        .iter()
348        .map(|p| match axis {
349            "x" => [p[1], p[2]],
350            "y" => [p[0], p[2]],
351            _ => [p[0], p[1]],
352        })
353        .collect()
354}

356/// Remove duplicate faces while preserving the first occurrence.
357fn dedup_faces(mut faces: Vec<Face>) -> Vec<Face> {
358    let mut seen = HashSet::new();
359    faces.retain(|f| seen.insert(face_key(f)));
360    faces
361}

363type FaceKey = (usize, usize, usize, usize, usize, usize, usize);

365/// Build a unique key for an exportable face record.
366fn face_record_key(record: &FaceRecord) -> FaceKey {
367    (
368        record.block_index,
369        record.imin,
370        record.jmin,
371        record.kmin,
372        record.imax,
373        record.jmax,
374        record.kmax,
375    )
376}

378/// Build a unique key directly from a `Face`.
379fn face_key(face: &Face) -> FaceKey {
380    (
381        face.block_index().unwrap_or(usize::MAX),
382        face.imin(),
383        face.jmin(),
384        face.kmin(),
385        face.imax(),
386        face.jmax(),
387        face.kmax(),
388    )

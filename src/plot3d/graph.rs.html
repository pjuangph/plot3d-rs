11/// Strategy for combining weights when multiple faces connect the same
12/// pair of blocks.
13#[derive(Copy, Clone, Debug, PartialEq, Eq)]
14pub enum WeightAggregate {
15    Sum,
16    Max,
17    Min,
18}
19
20/// Adjacency and edge-weight data for a block graph.
21#[derive(Clone, Debug)]
22pub struct BlockGraph {
23    /// Adjacency list: `adj_list[u]` is a sorted, deduplicated list of
24    /// neighbors for block `u`.
25    pub adj_list: Vec<Vec<usize>>,
26    /// Edge weights: `edge_weights[u][v]` = communication cost between `u` and `v`.
27    pub edge_weights: Vec<HashMap<usize, i64>>,
28}
29
30/// CSR (Compressed Sparse Row) representation of a block graph.
31#[derive(Clone, Debug)]
32pub struct CsrGraph {
33    pub xadj: Vec<i32>,
34    pub adjncy: Vec<i32>,
35    pub eweights: Vec<i32>,
36}

38/// Build a weighted graph from face-match data.
39///
40/// Each face match contributes a weight equal to the product of the face
41/// dimension ranges (`dI * dJ * dK`), representing the communication cost
42/// (number of shared nodes).
43pub fn build_weighted_graph_from_face_matches(
44    face_matches: &[FaceMatch],
45    n_blocks: usize,
46    aggregate: WeightAggregate,
47    ignore_self_matches: bool,
48) -> BlockGraph {
49    let mut pair_weight: HashMap<(usize, usize), i64> = HashMap::new();
50
51    for m in face_matches {
52        let i = m.block1.block_index;
53        let j = m.block2.block_index;
54        if ignore_self_matches && i == j {
55            continue;
56        }
57
58        let di = (m.block1.imax as i64 - m.block1.imin as i64).unsigned_abs().max(1);
59        let dj = (m.block1.jmax as i64 - m.block1.jmin as i64).unsigned_abs().max(1);
60        let dk = (m.block1.kmax as i64 - m.block1.kmin as i64).unsigned_abs().max(1);
61        let w = (di * dj * dk) as i64;
62
63        let (a, b) = if i < j { (i, j) } else { (j, i) };
64        let entry = pair_weight.entry((a, b)).or_insert(0);
65        match aggregate {
66            WeightAggregate::Sum => *entry += w,
67            WeightAggregate::Max => *entry = (*entry).max(w),
68            WeightAggregate::Min => {
69                if *entry == 0 {
70                    *entry = w;
71                } else {
72                    *entry = (*entry).min(w);
73                }
74            }
75        }
76    }
77
78    let mut adj_list: Vec<Vec<usize>> = vec![Vec::new(); n_blocks];
79    let mut edge_weights: Vec<HashMap<usize, i64>> = vec![HashMap::new(); n_blocks];
80
81    for (&(a, b), &w) in &pair_weight {
82        adj_list[a].push(b);
83        adj_list[b].push(a);
84        edge_weights[a].insert(b, w);
85        edge_weights[b].insert(a, w);
86    }
87
88    // Sort and deduplicate adjacency lists
89    for list in &mut adj_list {
90        list.sort_unstable();
91        list.dedup();
92    }
93
94    BlockGraph {
95        adj_list,
96        edge_weights,
97    }
98}

100/// Convert a [`BlockGraph`] to CSR arrays suitable for METIS.
101pub fn csr_from_block_graph(graph: &BlockGraph) -> CsrGraph {
102    let mut xadj: Vec<i32> = vec![0];
103    let mut adjncy: Vec<i32> = Vec::new();
104    let mut eweights: Vec<i32> = Vec::new();
105
106    let mut count: i32 = 0;
107    for u in 0..graph.adj_list.len() {
108        for &v in &graph.adj_list[u] {
109            adjncy.push(v as i32);
110            let w = graph.edge_weights[u].get(&v).copied().unwrap_or(1);
111            eweights.push(w as i32);
112            count += 1;
113        }
114        xadj.push(count);
115    }
116
117    CsrGraph {
118        xadj,
119        adjncy,
120        eweights,
121    }
122}

124/// Partition blocks using METIS graph partitioning.
125///
126/// Requires the `metis-partition` Cargo feature to be enabled.
127///
128/// Returns `(parts, graph)` where `parts[i]` is the 0-based partition ID
129/// for block `i`.
130#[cfg(feature = "metis-partition")]
131pub fn partition_from_face_matches(
132    face_matches: &[FaceMatch],
133    block_sizes: &[usize],
134    nparts: usize,
135    favor_blocksize: bool,
136    aggregate: WeightAggregate,
137    ignore_self_matches: bool,
138) -> Result<(Vec<i32>, BlockGraph), String> {
139    let n_blocks = block_sizes.len();
140    let graph = build_weighted_graph_from_face_matches(
141        face_matches,
142        n_blocks,
143        aggregate,
144        ignore_self_matches,
145    );
146    let csr = csr_from_block_graph(&graph);
147
148    let vwgt: Option<Vec<metis::Idx>> = if favor_blocksize {
149        Some(block_sizes.iter().map(|&s| s as metis::Idx).collect())
150    } else {
151        None
152    };
153
154    let mut part = vec![0 as metis::Idx; n_blocks];
155
156    // Build METIS graph and partition
157    let ncon: metis::Idx = 1;
158    let nparts_idx = nparts as metis::Idx;
159
160    let result = metis::Graph::new(ncon, nparts_idx, &csr.xadj, &csr.adjncy);
161    match result {
162        Ok(mut g) => {
163            if let Some(ref vw) = vwgt {
164                g = g.set_vwgt(vw);
165            }
166            g = g.set_adjwgt(&csr.eweights);
167            g.part_kway(&mut part)
168                .map_err(|e| format!("METIS partitioning failed: {:?}", e))?;
169        }
170        Err(e) => return Err(format!("Failed to create METIS graph: {:?}", e)),
171    }
172
173    Ok((part, graph))
174}

176/// Write ddcmp.dat and ddcmp_info.txt files for domain decomposition.
177///
178/// `parts` are 0-based internally, written 1-based in the file.
179pub fn write_ddcmp(
180    parts: &[i32],
181    block_sizes: &[usize],
182    graph: &BlockGraph,
183    filename: &str,
184) -> std::io::Result<()> {
185    let n_blocks = parts.len();
186    let n_proc = parts.iter().copied().max().unwrap_or(-1) + 1;
187    let n_isp = n_proc;
188
189    // Write ddcmp.dat
190    let path = Path::new(filename);
191    if let Some(parent) = path.parent() {
192        if !parent.as_os_str().is_empty() {
193            std::fs::create_dir_all(parent)?;
194        }
195    }
196
197    let mut f = std::fs::File::create(path)?;
198    writeln!(f, "{}", n_proc)?;
199    writeln!(f, "{}", n_isp)?;
200    writeln!(f, "{}", n_blocks)?;
201    for b_idx in 0..n_blocks {
202        writeln!(f, "{} {}", b_idx + 1, parts[b_idx] + 1)?;
203    }
204    for isp in 0..n_isp {
205        writeln!(f, "{} {}", isp + 1, isp)?;
206    }
207
208    // Compute statistics
209    let np = n_proc as usize;
210    let mut communication_work = vec![0i64; np];
211    let mut partition_edge_weights = vec![0i64; np];
212    let mut volume_nodes = vec![0usize; np];
213
214    for (b, &bsize) in block_sizes.iter().enumerate() {
215        let pid = parts[b] as usize;
216        volume_nodes[pid] += bsize;
217    }
218
219    for b in 0..block_sizes.len() {
220        let pid = parts[b] as usize;
221        for &nbr in &graph.adj_list[b] {
222            let nbr_pid = parts[nbr] as usize;
223            if nbr_pid != pid {
224                communication_work[pid] += 1;
225                partition_edge_weights[pid] +=
226                    graph.edge_weights[b].get(&nbr).copied().unwrap_or(1);
227            }
228        }
229    }
230
231    // Write ddcmp_info.txt
232    let info_path = path.with_file_name("ddcmp_info.txt");
233    let mut fi = std::fs::File::create(&info_path)?;
234    for i in 0..np {
235        let block_count = parts.iter().filter(|&&p| p == i as i32).count();
236        writeln!(fi, "Parition {} has {} blocks", i, block_count)?;
237    }
238    writeln!(fi, "Number of partitions/processors {}", n_proc)?;
239    for i in 0..np {
240        writeln!(
241            fi,
242            "Parition or processor {} has communication work {} edge_work {} volume_nodes {}",
243            i, communication_work[i], partition_edge_weights[i], volume_nodes[i]
244        )?;
245    }
246    let total_comm: i64 = communication_work.iter().sum();
247    let total_edge: i64 = partition_edge_weights.iter().sum();
248    writeln!(
249        fi,
250        "Total communication work {} Total edge_work {}",
251        total_comm, total_edge
252    )?;
253
254    Ok(())
255}

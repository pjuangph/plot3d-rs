14/// Merge two compatible blocks by aligning and stacking matched faces.
15///
16/// This is a fairly literal translation of the Python logic. Given a face pair
17/// reported by `find_matching_faces`, we permute/flip the second block so the
18/// matching faces line up, ensure both blocks increase in the same physical
19/// direction along the stacking axis, trim the overlapping slice, then
20/// concatenate before re-standardising orientation.
21///
22/// Returns a merged block, or `block1` unchanged when no compatible merge can
23/// be produced (mirroring the Python helper's behaviour).
24pub fn combine_2_blocks_mixed_pairing(block1: &Block, block2: &Block, tol: f64) -> Block {
25    let Some((face1, face2, (flip_ud, flip_lr))) = find_matching_faces(block1, block2, tol) else {
26        return block1.clone();
27    };
28
29    let (axis1, _dir1) = face_axis_info(face1);
30    let (axis2, _dir2) = face_axis_info(face2);
31
32    let mut base = block1.clone();
33    let mut other = block2.clone();
34    let mut face_label = face2.to_string();
35
36    if axis1 != axis2 {
37        let mut perm = [0usize, 1, 2];
38        perm.swap(axis1, axis2);
39        other = permute_block_axes(&other, perm);
40        face_label = remap_face_label(&face_label, perm);
41    }
42
43    let stack_axis = axis1;
44    let target_dims = [base.imax, base.jmax, base.kmax];
45    let (other_aligned, perm_opt) = match align_cross_sections(other, target_dims, stack_axis) {
46        Some(result) => result,
47        None => return block1.clone(),
48    };
49    other = other_aligned;
50    if let Some(perm_align) = perm_opt {
51        face_label = remap_face_label(&face_label, perm_align);
52    }
53
54    let (flip_ud_axis, flip_lr_axis) = flip_axes_for_face(&face_label);
55    other = apply_face_flips(&other, flip_ud_axis, flip_lr_axis, flip_ud, flip_lr);
56
57    // Match Python logic: choose the dominant coordinate component based on
58    // block1, then compare block2 on that same component to decide flipping.
59    let steps1 = component_steps(&base, stack_axis);
60    let dominant_idx = argmax_abs(&steps1);
61    let step1 = steps1[dominant_idx];
62    let step2 = component_steps(&other, stack_axis)[dominant_idx];
63
64    if step1.signum() != 0.0 && step2.signum() != 0.0 && step1.signum() != step2.signum() {
65        base = flip_block_axis(&base, stack_axis);
66    }
67
68    let drop_first = face_label.ends_with("min");
69    let trimmed_other = trim_block_along_axis(&other, stack_axis, drop_first);
70
71    let merged = if drop_first {
72        match concat_blocks_along_axis(&base, &trimmed_other, stack_axis) {
73            Some(block) => block,
74            None => return block1.clone(),
75        }
76    } else {
77        match concat_blocks_along_axis(&trimmed_other, &base, stack_axis) {
78            Some(block) => block,
79            None => return block1.clone(),
80        }
81    };
82
83    standardize_block_orientation(&merged)
84}

86/// Attempt to merge as many blocks as possible from an initial set.
87///
88/// Starting from the provided block list (typically ≤ 8 items), this makes
89/// repeated passes trying to merge any pair whose faces match. The search is
90/// greedy, mirroring the original Python helper: once a pair is merged it is
91/// replaced by the new block and the pass restarts until no further reductions
92/// occur or `max_tries` is hit.
93pub fn combine_blocks_mixed_pairs(blocks: &[Block], tol: f64, max_tries: usize) -> CombinedBlocks {
94    let mut merged_blocks: Vec<Block> = blocks.to_vec();
95    let mut tries = 0usize;
96
97    while merged_blocks.len() > 1 && tries < max_tries {
98        let mut new_merged: Vec<Block> = Vec::new();
99        let mut skip: HashSet<usize> = HashSet::new();
100        let mut any_merge = false;
101        let mut i = 0usize;
102
103        while i < merged_blocks.len() {
104            if skip.contains(&i) {
105                i += 1;
106                continue;
107            }
108
109            let blk_a = merged_blocks[i].clone();
110            let mut merged: Option<Block> = None;
111            let mut partner_idx: Option<usize> = None;
112
113            for j in (i + 1)..merged_blocks.len() {
114                if skip.contains(&j) {
115                    continue;
116                }
117
118                if find_matching_faces(&merged_blocks[i], &merged_blocks[j], tol).is_some() {
119                    let candidate =
120                        combine_2_blocks_mixed_pairing(&merged_blocks[i], &merged_blocks[j], tol);
121                    merged = Some(candidate);
122                    partner_idx = Some(j);
123                    break;
124                }
125            }
126
127            if let Some(block) = merged {
128                new_merged.push(block);
129                skip.insert(i);
130                if let Some(j) = partner_idx {
131                    skip.insert(j);
132                }
133                any_merge = true;
134            } else {
135                new_merged.push(blk_a);
136                skip.insert(i);
137            }
138
139            i += 1;
140        }
141
142        for k in 0..merged_blocks.len() {
143            if !skip.contains(&k) {
144                new_merged.push(merged_blocks[k].clone());
145            }
146        }
147
148        if !any_merge {
149            break;
150        }
151
152        merged_blocks = new_merged;
153        tries += 1;
154    }
155
156    let used_indices = (0..blocks.len()).collect();
157    (merged_blocks, used_indices)
158}

160/// Merge all discoverable n×n×n cube groupings using connectivity data.
161///
162/// `connectivities` should be the face matches returned by `connectivity`/
163/// `connectivity_fast`. The routine builds a graph, performs BFS to locate
164/// candidate cube groupings (`cube_size^3` nodes), merges their blocks using
165/// `combine_blocks_mixed_pairs`, and keeps track of which original indices fed
166/// each merged component.
167pub fn combine_nxnxn_cubes_mixed_pairs(
168    blocks: &[Block],
169    connectivities: &[FaceMatch],
170    cube_size: usize,
171    tol: Option<f64>,
172) -> Vec<(Block, HashSet<usize>)> {
173    let tol = tol.unwrap_or(1e-8);
174    if cube_size == 0 {
175        return Vec::new();
176    }
177    let target_size = cube_size.pow(3);
178
179    let graph = build_connectivity_graph(connectivities);
180    let mut used: HashSet<usize> = HashSet::new();
181    let mut remaining: Vec<usize> = (0..blocks.len()).collect();
182    let mut merged_groups = Vec::new();
183
184    loop {
185        let before_len = remaining.len();
186        let mut merged_this_round = false;
187        let mut new_used: HashSet<usize> = HashSet::new();
188
189        let mut idx = 0usize;
190        while idx < remaining.len() {
191            let seed = remaining[idx];
192            if used.contains(&seed) {
193                idx += 1;
194                continue;
195            }
196
197            let group_opt = find_nxnxn_group(seed, &graph, &used, target_size);
198            let Some(group) = group_opt else {
199                idx += 1;
200                continue;
201            };
202
203            if !group.is_disjoint(&new_used) {
204                idx += 1;
205                continue;
206            }
207
208            let mut sorted_group: Vec<usize> = group.iter().copied().collect();
209            sorted_group.sort_unstable();
210
211            let group_blocks: Vec<Block> =
212                sorted_group.iter().map(|&i| blocks[i].clone()).collect();
213            // Follow Python default: attempt several passes (4) when merging
214            // a candidate group, instead of tying tries to `cube_size`.
215            let (partial_merges, local_indices) =
216                combine_blocks_mixed_pairs(&group_blocks, tol, 4);
217
218            let index_mapping: HashMap<usize, usize> = sorted_group
219                .iter()
220                .enumerate()
221                .map(|(local, &global)| (local, global))
222                .collect();
223
224            for merged_block in partial_merges {
225                let mut merged_ids = HashSet::new();
226                for &local in &local_indices {
227                    if let Some(global) = index_mapping.get(&local) {
228                        merged_ids.insert(*global);
229                    }
230                }
231                if merged_ids.is_empty() {
232                    continue;
233                }
234                new_used.extend(&merged_ids);
235                merged_groups.push((merged_block, merged_ids));
236            }
237
238            merged_this_round = true;
239            remaining.retain(|idx| !new_used.contains(idx));
240            idx = 0;
241        }
242
243        used.extend(&new_used);
244
245        if !merged_this_round || remaining.len() == before_len {
246            for idx in remaining {
247                if used.contains(&idx) {
248                    continue;
249                }
250                let mut set = HashSet::new();
251                set.insert(idx);
252                merged_groups.push((blocks[idx].clone(), set));
253            }
254            break;
255        }
256    }
257
258    merged_groups
259}

261/// Perform a breadth-first search from `seed` to find a cube-sized group that
262/// has not yet been merged or marked as used.
263fn find_nxnxn_group(
264    seed: usize,
265    graph: &HashMap<usize, HashSet<usize>>,
266    used: &HashSet<usize>,
267    target_size: usize,
268) -> Option<HashSet<usize>> {
269    let mut visited = HashSet::new();
270    let mut queue = VecDeque::new();
271    queue.push_back(seed);
272
273    while let Some(idx) = queue.pop_front() {
274        if visited.contains(&idx) || used.contains(&idx) {
275            continue;
276        }
277        visited.insert(idx);
278        if visited.len() == target_size {
279            break;
280        }
281        if let Some(neighbors) = graph.get(&idx) {
282            for &nbr in neighbors {
283                if !visited.contains(&nbr) && !used.contains(&nbr) {
284                    queue.push_back(nbr);
285                }
286            }
287        }
288    }
289
290    if visited.len() == target_size {
291        Some(visited)
292    } else {
293        None
294    }
295}

297/// Map a face name to its corresponding axis (0/1/2) and sign direction.
298fn face_axis_info(face: &str) -> (usize, i32) {
299    match face {
300        "imin" => (0, -1),
301        "imax" => (0, 1),
302        "jmin" => (1, -1),
303        "jmax" => (1, 1),
304        "kmin" => (2, -1),
305        "kmax" => (2, 1),
306        _ => (0, 0),
307    }

310/// Reorder the block axes according to `perm`, returning a new block.
311fn permute_block_axes(block: &Block, perm: [usize; 3]) -> Block {
312    let dims = [block.imax, block.jmax, block.kmax];
313    let new_dims = [dims[perm[0]], dims[perm[1]], dims[perm[2]]];
314    let mut x = vec![0.0; new_dims[0] * new_dims[1] * new_dims[2]];
315    let mut y = x.clone();
316    let mut z = x.clone();
317
318    for i_new in 0..new_dims[0] {
319        for j_new in 0..new_dims[1] {
320            for k_new in 0..new_dims[2] {
321                let mut old = [0usize; 3];
322                old[perm[0]] = i_new;
323                old[perm[1]] = j_new;
324                old[perm[2]] = k_new;
325                let (vx, vy, vz) = block.xyz(old[0], old[1], old[2]);
326                let idx = linear_index(new_dims, [i_new, j_new, k_new]);
327                x[idx] = vx;
328                y[idx] = vy;
329                z[idx] = vz;
330            }
331        }
332    }
333
334    Block::new(new_dims[0], new_dims[1], new_dims[2], x, y, z)
335}

337/// Ensure the non-stacking axes of `block` match `target_dims`, optionally swapping them.
338fn align_cross_sections(
339    block: Block,
340    target_dims: [usize; 3],
341    stack_axis: usize,
342) -> Option<(Block, Option<[usize; 3]>)> {
343    let dims = [block.imax, block.jmax, block.kmax];
344    let cross_axes: Vec<usize> = (0..3).filter(|&ax| ax != stack_axis).collect();
345    if cross_axes.len() != 2 {
346        return Some((block, None));
347    }
348    let axis_a = cross_axes[0];
349    let axis_b = cross_axes[1];
350    let aligned = dims[axis_a] == target_dims[axis_a] && dims[axis_b] == target_dims[axis_b];
351    if aligned {
352        return Some((block, None));
353    }
354    None
355}

358fn apply_face_flips(

375/// Compute step magnitudes for X, Y, Z along `axis`.
376fn component_steps(block: &Block, axis: usize) -> [f64; 3] {
377    [
378        coordinate_step(block, axis, 0),
379        coordinate_step(block, axis, 1),
380        coordinate_step(block, axis, 2),
381    ]

384fn argmax_abs(vals: &[f64; 3]) -> usize {
385    let mut best = 0usize;
386    let mut best_abs = vals[0].abs();
387    for (i, v) in vals.iter().enumerate().skip(1) {
388        let a = v.abs();
389        if a > best_abs {
390            best = i;
391            best_abs = a;
392        }
393    }
394    best
395}

397/// Compute the signed step between the first and last plane along `axis`
398/// for the requested coordinate component (0 → X, 1 → Y, 2 → Z).
399fn coordinate_step(block: &Block, axis: usize, component: usize) -> f64 {
400    let dims = [block.imax, block.jmax, block.kmax];
401    if dims[axis] <= 1 {
402        return 0.0;
403    }
404    let mut start = [dims[0] / 2, dims[1] / 2, dims[2] / 2];
405    let mut end = start;
406    start[axis] = 0;
407    end[axis] = dims[axis] - 1;
408    let start_val = component_value(block, start, component);
409    let end_val = component_value(block, end, component);
410    end_val - start_val
411}

413fn component_value(block: &Block, idx: [usize; 3], component: usize) -> f64 {
414    match component {
415        0 => block.x[linear_index([block.imax, block.jmax, block.kmax], idx)],
416        1 => block.y[linear_index([block.imax, block.jmax, block.kmax], idx)],
417        _ => block.z[linear_index([block.imax, block.jmax, block.kmax], idx)],
418    }
419}

421/// Create a copy of the block with the specified axis reversed.
422fn flip_block_axis(block: &Block, axis: usize) -> Block {
423    let dims = [block.imax, block.jmax, block.kmax];
424    let mut x = vec![0.0; block.npoints()];
425    let mut y = x.clone();
426    let mut z = x.clone();
427
428    for i in 0..dims[0] {
429        for j in 0..dims[1] {
430            for k in 0..dims[2] {
431                let mut src = [i, j, k];
432                src[axis] = dims[axis] - 1 - src[axis];
433                let (vx, vy, vz) = block.xyz(src[0], src[1], src[2]);
434                let idx = linear_index(dims, [i, j, k]);
435                x[idx] = vx;
436                y[idx] = vy;
437                z[idx] = vz;
438            }
439        }
440    }
441
442    Block::new(dims[0], dims[1], dims[2], x, y, z)
443}

445/// Drop the overlapping slice along `axis` (front or back) before concatenation.
446fn trim_block_along_axis(block: &Block, axis: usize, drop_first: bool) -> Block {
447    let dims = [block.imax, block.jmax, block.kmax];
448    if dims[axis] <= 1 {
449        return block.clone();
450    }
451    let mut new_dims = dims;
452    new_dims[axis] -= 1;
453    let mut x = vec![0.0; new_dims[0] * new_dims[1] * new_dims[2]];
454    let mut y = x.clone();
455    let mut z = x.clone();
456
457    for i in 0..new_dims[0] {
458        for j in 0..new_dims[1] {
459            for k in 0..new_dims[2] {
460                let mut src = [i, j, k];
461                if drop_first {
462                    src[axis] += 1;
463                }
464                let (vx, vy, vz) = block.xyz(src[0], src[1], src[2]);
465                let idx = linear_index(new_dims, [i, j, k]);
466                x[idx] = vx;
467                y[idx] = vy;
468                z[idx] = vz;
469            }
470        }
471    }
472
473    Block::new(new_dims[0], new_dims[1], new_dims[2], x, y, z)
474}

476/// Concatenate two blocks along `axis`, assuming matching cross-sections.
477/// Returns `None` when the non-stacking dimensions do not align.
478fn concat_blocks_along_axis(a: &Block, b: &Block, axis: usize) -> Option<Block> {
479    let dims_a = [a.imax, a.jmax, a.kmax];
480    let dims_b = [b.imax, b.jmax, b.kmax];
481    let mut new_dims = dims_a;
482    new_dims[axis] += dims_b[axis];
483
484    for idx in 0..3 {
485        if idx != axis && dims_a[idx] != dims_b[idx] {
486            return None;
487        }
488    }
489
490    let total = new_dims[0] * new_dims[1] * new_dims[2];
491    let mut x = vec![0.0; total];
492    let mut y = x.clone();
493    let mut z = x.clone();
494
495    for i in 0..new_dims[0] {
496        for j in 0..new_dims[1] {
497            for k in 0..new_dims[2] {
498                let idx_new = linear_index(new_dims, [i, j, k]);
499                let coord = if coordinate_from_block(dims_a, axis, [i, j, k]) {
500                    let src = [i, j, k];
501                    a.xyz(src[0], src[1], src[2])
502                } else {
503                    let mut src = [i, j, k];
504                    src[axis] -= dims_a[axis];
505                    b.xyz(src[0], src[1], src[2])
506                };
507                x[idx_new] = coord.0;
508                y[idx_new] = coord.1;
509                z[idx_new] = coord.2;
510            }
511        }
512    }
513
514    Some(Block::new(new_dims[0], new_dims[1], new_dims[2], x, y, z))
515}

517fn coordinate_from_block(dims_a: [usize; 3], axis: usize, idx: [usize; 3]) -> bool {

521fn linear_index(dims: [usize; 3], idx: [usize; 3]) -> usize {
522    (idx[2] * dims[1] + idx[1]) * dims[0] + idx[0]

525fn flip_axes_for_face(face: &str) -> (usize, usize) {
526    match face.chars().next().map(|c| c.to_ascii_lowercase()) {
527        Some('i') => (1, 2),
528        Some('j') => (0, 2),
529        Some('k') => (0, 1),
530        _ => (1, 2),
531    }

534fn remap_face_label(face: &str, perm: [usize; 3]) -> String {
535    let mut chars = face.chars();
536    let Some(axis_char) = chars.next() else {
537        return face.to_string();
538    };
539    let remainder: String = chars.collect();
540    let orig_axis = match axis_char.to_ascii_lowercase() {
541        'i' => 0,
542        'j' => 1,
543        'k' => 2,
544        _ => return face.to_string(),
545    };
546    let new_axis_idx = perm
547        .iter()
548        .position(|&old_axis| old_axis == orig_axis)
549        .unwrap_or(orig_axis);
550    let mut new_axis_char = match new_axis_idx {
551        0 => 'i',
552        1 => 'j',
553        2 => 'k',
554        _ => axis_char.to_ascii_lowercase(),
555    };
556    if axis_char.is_ascii_uppercase() {
557        new_axis_char = new_axis_char.to_ascii_uppercase();
558    }
559    format!("{new_axis_char}{remainder}")
560}

29/// Compact record describing a face on a particular block.
30#[derive(Clone, Debug, Serialize)]
31pub struct FaceRecord {
32    pub block_index: usize,
33    pub imin: usize,
34    pub jmin: usize,
35    pub kmin: usize,
36    pub imax: usize,
37    pub jmax: usize,
38    pub kmax: usize,
39    pub id: Option<usize>,
40}
41
42impl FaceRecord {
43    /// Build a corner description from matching points.
44    ///
45    /// * `block_index` – Owning block index.
46    /// * `points` – Matched nodes.
47    /// * `first` – If `true` we use the indices from block1; otherwise block2.
48    ///
49    /// Returns `None` when `points` is empty.
50    fn from_match_points(block_index: usize, points: &[MatchPoint], first: bool) -> Option<Self> {
51        if points.is_empty() {
52            return None;
53        }
54        let imin = points
55            .iter()
56            .map(|p| if first { p.i1 } else { p.i2 })
57            .min()?;
58        let jmin = points
59            .iter()
60            .map(|p| if first { p.j1 } else { p.j2 })
61            .min()?;
62        let kmin = points
63            .iter()
64            .map(|p| if first { p.k1 } else { p.k2 })
65            .min()?;
66        let imax = points
67            .iter()
68            .map(|p| if first { p.i1 } else { p.i2 })
69            .max()?;
70        let jmax = points
71            .iter()
72            .map(|p| if first { p.j1 } else { p.j2 })
73            .max()?;
74        let kmax = points
75            .iter()
76            .map(|p| if first { p.k1 } else { p.k2 })
77            .max()?;
78        Some(Self {
79            block_index,
80            imin,
81            jmin,
82            kmin,
83            imax,
84            jmax,
85            kmax,
86            id: None,
87        })
88    }
89
90    /// Construct a record from a Face instance.
91    pub fn from_face(face: &crate::block_face_functions::Face) -> Self {
92        Self {
93            block_index: face.block_index().unwrap_or(usize::MAX),
94            imin: face.imin(),
95            jmin: face.jmin(),
96            kmin: face.kmin(),
97            imax: face.imax(),
98            jmax: face.jmax(),
99            kmax: face.kmax(),
100            id: face.id(),
101        }
102    }
103
104    /// Scale the index ranges by `factor`.
105    pub fn scale_indices(&mut self, factor: usize) {
106        if factor <= 1 {
107            return;
108        }
109        self.imin *= factor;
110        self.jmin *= factor;
111        self.kmin *= factor;
112        self.imax *= factor;
113        self.jmax *= factor;
114        self.kmax *= factor;
115    }
116
117    /// Reduce the index ranges by `divisor`.
118    pub fn divide_indices(&mut self, divisor: usize) {
119        if divisor <= 1 {
120            return;
121        }
122        self.imin /= divisor;
123        self.jmin /= divisor;
124        self.kmin /= divisor;
125        self.imax /= divisor;
126        self.jmax /= divisor;
127        self.kmax /= divisor;
128    }
129
130    /// Reconstruct a Face from this record using the provided blocks.
131    pub fn to_face(
132        &self,
133        blocks: &[crate::block::Block],
134    ) -> Option<crate::block_face_functions::Face> {
135        let block = blocks.get(self.block_index)?;
136        let mut face = crate::block_face_functions::create_face_from_diagonals(
137            block, self.imin, self.jmin, self.kmin, self.imax, self.jmax, self.kmax,
138        );
139        face.set_block_index(self.block_index);
140        if let Some(id) = self.id {
141            face.set_id(id);
142        }
143        Some(face)
144    }
145}

242/// Structured-grid node on a face, capturing indices and XYZ coordinate.
243#[derive(Clone, Debug)]
244struct FaceNode {
245    i: usize,
246    j: usize,
247    k: usize,
248    coord: [f64; 3],
249}
250
251/// Enumerate all nodes that belong to `face` on `block`.
252///
253/// # Arguments
254/// * `face` - Face whose nodes should be sampled.
255/// * `block` - Parent block providing Cartesian coordinates.
256///
257/// # Returns
258/// Vector of [`FaceNode`] containing structured indices `(i, j, k)` and the
259/// corresponding XYZ coordinate.
260fn face_nodes(face: &Face, block: &Block) -> Vec<FaceNode> {
261    let mut nodes = Vec::new();
262    let i_vals: Vec<usize> = if face.imin() == face.imax() {
263        vec![face.imin()]
264    } else {
265        (face.imin()..=face.imax()).collect()
266    };
267    let j_vals: Vec<usize> = if face.jmin() == face.jmax() {
268        vec![face.jmin()]
269    } else {
270        (face.jmin()..=face.jmax()).collect()
271    };
272    let k_vals: Vec<usize> = if face.kmin() == face.kmax() {
273        vec![face.kmin()]
274    } else {
275        (face.kmin()..=face.kmax()).collect()
276    };
277    for &i in &i_vals {
278        for &j in &j_vals {
279            for &k in &k_vals {
280                if !(i < block.imax && j < block.jmax && k < block.kmax) {
281                    continue;
282                }
283                let (x, y, z) = block.xyz(i, j, k);
284                nodes.push(FaceNode {
285                    i,
286                    j,
287                    k,
288                    coord: [x, y, z],
289                });
290            }
291        }
292    }
293    nodes
294}

296/// Locate the node whose coordinate is within `tol` of `target`.
297///
298/// Returns the first node that meets the tolerance, preferring the closest
299/// distance. When no node matches, `None` is returned.
300fn find_closest_node<'a>(
301    nodes: &'a [FaceNode],
302    target: [f64; 3],
303    tol: f64,
304) -> Option<&'a FaceNode> {
305    let mut best: Option<(&FaceNode, f64)> = None;
306    for node in nodes {
307        let dx = node.coord[0] - target[0];
308        let dy = node.coord[1] - target[1];
309        let dz = node.coord[2] - target[2];
310        let dist = (dx * dx + dy * dy + dz * dz).sqrt();
311        if dist <= tol {
312            match best {
313                Some((_, best_dist)) if dist >= best_dist => {}
314                _ => best = Some((node, dist)),
315            }
316        }
317    }
318    best.map(|(node, _)| node)
319}

321/// Check whether the coincident nodes degenerate to an edge contact.
322fn is_edge(points: &[MatchPoint]) -> bool {
323    if points.is_empty() {
324        return false;
325    }
326    let min_i1 = points.iter().map(|p| p.i1).min().unwrap();
327    let max_i1 = points.iter().map(|p| p.i1).max().unwrap();
328    let min_j1 = points.iter().map(|p| p.j1).min().unwrap();
329    let max_j1 = points.iter().map(|p| p.j1).max().unwrap();
330    let min_k1 = points.iter().map(|p| p.k1).min().unwrap();
331    let max_k1 = points.iter().map(|p| p.k1).max().unwrap();
332
333    let mut edge_matches = 0;
334    if min_i1 == max_i1 {
335        edge_matches += 1;
336    }
337    if min_j1 == max_j1 {
338        edge_matches += 1;
339    }
340    if min_k1 == max_k1 {
341        edge_matches += 1;
342    }
343    edge_matches >= 2
344}

346/// Filter matches so the provided key advances monotonically by 1.
347fn filter_block_increasing(
348    points: &[MatchPoint],
349    key: fn(&MatchPoint) -> usize,
350) -> Vec<MatchPoint> {
351    if points.is_empty() {
352        return Vec::new();
353    }
354    let mut unique_vals: Vec<usize> = points.iter().map(key).collect();
355    unique_vals.sort_unstable();
356    unique_vals.dedup();
357    if unique_vals.len() <= 1 {
358        return Vec::new();
359    }
360    let mut keep: HashSet<usize> = HashSet::new();
361    for window in unique_vals.windows(2) {
362        if window[1] == window[0] + 1 {
363            keep.insert(window[0]);
364        }
365    }
366    if let (Some(last), Some(prev)) = (
367        unique_vals.last(),
368        unique_vals.get(unique_vals.len().saturating_sub(2)),
369    ) {
370        if *last == *prev + 1 {
371            keep.insert(*last);
372        }
373    }
374    points
375        .iter()
376        .filter(|p| keep.contains(&key(p)))
377        .cloned()
378        .collect()
379}

381/// Enforce monotonic progression along the non-constant axes of each face.
382fn apply_axis_filters(points: Vec<MatchPoint>, face1: &Face, face2: &Face) -> Vec<MatchPoint> {
383    let mut filtered = points;
384    match face1.const_axis() {
385        Some(crate::block_face_functions::FaceAxis::I) => {
386            filtered = filter_block_increasing(&filtered, |p| p.j1);
387            filtered = filter_block_increasing(&filtered, |p| p.k1);
388        }
389        Some(crate::block_face_functions::FaceAxis::J) => {
390            filtered = filter_block_increasing(&filtered, |p| p.i1);
391            filtered = filter_block_increasing(&filtered, |p| p.k1);
392        }
393        Some(crate::block_face_functions::FaceAxis::K) => {
394            filtered = filter_block_increasing(&filtered, |p| p.i1);
395            filtered = filter_block_increasing(&filtered, |p| p.j1);
396        }
397        None => {}
398    }
399    match face2.const_axis() {
400        Some(crate::block_face_functions::FaceAxis::I) => {
401            filtered = filter_block_increasing(&filtered, |p| p.j2);
402            filtered = filter_block_increasing(&filtered, |p| p.k2);
403        }
404        Some(crate::block_face_functions::FaceAxis::J) => {
405            filtered = filter_block_increasing(&filtered, |p| p.i2);
406            filtered = filter_block_increasing(&filtered, |p| p.k2);
407        }
408        Some(crate::block_face_functions::FaceAxis::K) => {
409            filtered = filter_block_increasing(&filtered, |p| p.i2);
410            filtered = filter_block_increasing(&filtered, |p| p.j2);
411        }
412        None => {}
413    }
414    filtered
415}

417/// Build subfaces produced by the intersection region.
418fn create_split_faces(
419    face: &Face,
420    block: &Block,
421    points: &[MatchPoint],
422    use_block1: bool,
423) -> Vec<Face> {
424    if points.is_empty() {
425        return Vec::new();
426    }
427    let (imin, imax, jmin, jmax, kmin, kmax) = if use_block1 {
428        (
429            points.iter().map(|p| p.i1).min().unwrap(),
430            points.iter().map(|p| p.i1).max().unwrap(),
431            points.iter().map(|p| p.j1).min().unwrap(),
432            points.iter().map(|p| p.j1).max().unwrap(),
433            points.iter().map(|p| p.k1).min().unwrap(),
434            points.iter().map(|p| p.k1).max().unwrap(),
435        )
436    } else {
437        (
438            points.iter().map(|p| p.i2).min().unwrap(),
439            points.iter().map(|p| p.i2).max().unwrap(),
440            points.iter().map(|p| p.j2).min().unwrap(),
441            points.iter().map(|p| p.j2).max().unwrap(),
442            points.iter().map(|p| p.k2).min().unwrap(),
443            points.iter().map(|p| p.k2).max().unwrap(),
444        )
445    };
446    let degeneracy =
447        usize::from(imin == imax) + usize::from(jmin == jmax) + usize::from(kmin == kmax);
448    if degeneracy != 1 {
449        return Vec::new();
450    }
451    let mut split = split_face(face, block, imin, jmin, kmin, imax, jmax, kmax);
452    for f in &mut split {
453        if let Some(idx) = face.block_index() {
454            f.set_block_index(idx);
455        }
456        if let Some(id) = face.id() {
457            f.set_id(id);
458        }
459    }
460    split
461}

463/// Compute the coincident nodes between two faces on separate blocks.
464///
465/// # Arguments
466/// * `face1` - Candidate face on `block1`.
467/// * `face2` - Candidate face on `block2`.
468/// * `block1` / `block2` - Parent blocks.
469/// * `tol` - Euclidean tolerance for node matching.
470///
471/// # Returns
472/// Tuple containing:
473/// 1. List of [`MatchPoint`]s.
474/// 2. Split faces generated on `block1`.
475/// 3. Split faces generated on `block2`.
476pub fn get_face_intersection(
477    face1: &Face,
478    face2: &Face,
479    block1: &Block,
480    block2: &Block,
481    tol: f64,
482) -> (Vec<MatchPoint>, Vec<Face>, Vec<Face>) {
483    let nodes1 = face_nodes(face1, block1);
484    let nodes2 = face_nodes(face2, block2);
485    let mut matches = Vec::new();
486    for node1 in &nodes1 {
487        if let Some(node2) = find_closest_node(&nodes2, node1.coord, tol) {
488            matches.push(MatchPoint {
489                i1: node1.i,
490                j1: node1.j,
491                k1: node1.k,
492                i2: node2.i,
493                j2: node2.j,
494                k2: node2.k,
495            });
496        }
497    }
498    if matches.len() < 4 || is_edge(&matches) {
499        return (Vec::new(), Vec::new(), Vec::new());
500    }
501    let matches = apply_axis_filters(matches, face1, face2);
502    if matches.len() < 4 {
503        return (Vec::new(), Vec::new(), Vec::new());
504    }
505
506    let split_faces1 = create_split_faces(face1, block1, &matches, true);
507    let split_faces2 = create_split_faces(face2, block2, &matches, false);
508    (matches, split_faces1, split_faces2)
509}

511/// Recursively match all faces between a pair of blocks.
512///
513/// # Arguments
514/// * `block1` / `block2` - Blocks to compare.
515/// * `block1_outer` / `block2_outer` - Mutable outer-face lists that will be
516///   updated in-place as faces are split.
517/// * `tol` - Node matching tolerance.
518///
519/// # Returns
520/// Collection of match-point arrays, one entry per detected interface.
521pub fn find_matching_blocks(
522    block1: &Block,
523    block2: &Block,
524    block1_outer: &mut Vec<Face>,
525    block2_outer: &mut Vec<Face>,
526    tol: f64,
527) -> Vec<Vec<MatchPoint>> {
528    let mut matches = Vec::new();
529    let mut i = 0;
530    'outer: while i < block1_outer.len() {
531        let mut j = 0;
532        while j < block2_outer.len() {
533            let face1 = block1_outer[i].clone();
534            let face2 = block2_outer[j].clone();
535            let (match_points, split1, split2) =
536                get_face_intersection(&face1, &face2, block1, block2, tol);
537            if !match_points.is_empty() {
538                matches.push(match_points.clone());
539
540                block1_outer.remove(i);
541                block2_outer.remove(j);
542                block1_outer.extend(split1);
543                block2_outer.extend(split2);
544                i = 0;
545                continue 'outer;
546            } else {
547                j += 1;
548            }
549        }
550        i += 1;
551    }
552    matches
553}

555/// Generate block index pairs using centroid distance ordering.
556///
557/// # Arguments
558/// * `blocks` - All blocks in the assembly.
559/// * `nearest_nblocks` - Number of closest neighbours to include for each block.
560///
561/// # Returns
562/// Vector of ordered `(i, j)` index pairs.
563fn combinations_of_nearest_blocks(blocks: &[Block], nearest_nblocks: usize) -> Vec<(usize, usize)> {
564    let centroids: Vec<(f64, f64, f64)> = blocks.iter().map(Block::centroid).collect();
565    let mut combos = Vec::new();
566    for (i, &ci) in centroids.iter().enumerate() {
567        let mut distances: Vec<(usize, f64)> = centroids
568            .iter()
569            .enumerate()
570            .filter(|(j, _)| *j != i)
571            .map(|(j, cj)| {
572                let dx = ci.0 - cj.0;
573                let dy = ci.1 - cj.1;
574                let dz = ci.2 - cj.2;
575                (j, (dx * dx + dy * dy + dz * dz).sqrt())
576            })
577            .collect();
578        distances.sort_by(|a, b| a.1.partial_cmp(&b.1).unwrap());
579        for (j, dist) in distances.into_iter().take(nearest_nblocks) {
580            if dist.is_finite() {
581                combos.push((i, j));
582            }
583        }
584    }
585    combos
586}

588/// Connectivity computation performed on GCD-reduced blocks.
589///
590/// # Arguments
591/// * `blocks` - Original block list. Each block is down-sampled by the
592///   smallest index GCD across the set.
593///
594/// # Returns
595/// Tuple `(matches, outer_faces)` where `matches` enumerates face interfaces
596/// and `outer_faces` records the remaining external surfaces at the original
597/// resolution.
598pub fn connectivity_fast(blocks: &[Block]) -> (Vec<FaceMatch>, Vec<FaceRecord>) {
599    let mut gcd_array = Vec::with_capacity(blocks.len());
600    for block in blocks {
601        let gcd = gcd_three(block.imax - 1, block.jmax - 1, block.kmax - 1);
602        gcd_array.push(gcd);
603    }
604    let gcd_to_use = gcd_array.into_iter().min().unwrap_or(1).max(1);
605    let reduced_blocks = crate::block_face_functions::reduce_blocks(blocks, gcd_to_use);
606    let (mut matches, mut outer_faces) = connectivity(&reduced_blocks);
607    // Scale back to origional size
608    for face in &mut matches {
609        face.block1.imin *= gcd_to_use;
610        face.block1.jmin *= gcd_to_use;
611        face.block1.kmin *= gcd_to_use;
612        face.block1.imax *= gcd_to_use;
613        face.block1.jmax *= gcd_to_use;
614        face.block1.kmax *= gcd_to_use;
615
616        face.block2.imin *= gcd_to_use;
617        face.block2.jmin *= gcd_to_use;
618        face.block2.kmin *= gcd_to_use;
619        face.block2.imax *= gcd_to_use;
620        face.block2.jmax *= gcd_to_use;
621        face.block2.kmax *= gcd_to_use;
622    }
623    for face in &mut outer_faces {
624        face.imin *= gcd_to_use;
625        face.jmin *= gcd_to_use;
626        face.kmin *= gcd_to_use;
627        face.imax *= gcd_to_use;
628        face.jmax *= gcd_to_use;
629        face.kmax *= gcd_to_use;
630    }
631    (matches, outer_faces)
632}

634/// Determine face-to-face connectivity and exterior faces for all blocks.
635///
636/// # Arguments
637/// * `blocks` - Full-resolution blocks to analyse.
638///
639/// # Returns
640/// Tuple `(matches, outer_faces)` representing matched interfaces and the
641/// formatted list of outer faces.
642pub fn connectivity(blocks: &[Block]) -> (Vec<FaceMatch>, Vec<FaceRecord>) {
643    let mut block_outer_faces: Vec<Vec<Face>> = blocks
644        .iter()
645        .enumerate()
646        .map(|(idx, block)| {
647            let (faces, _) = get_outer_faces(block);
648            faces
649                .into_iter()
650                .map(|mut f| {
651                    f.set_block_index(idx);
652                    f
653                })
654                .collect()
655        })
656        .collect();
657
658    let combos = combinations_of_nearest_blocks(blocks, 6);
659    let mut matches = Vec::new();
660    let mut matches_to_remove: HashSet<(usize, usize, usize, usize, usize, usize, usize)> =
661        HashSet::new();
662
663    for (i, j) in combos {
664        if i == j {
665            continue;
666        }
667        let (left, right) = if i < j {
668            let (left, right) = block_outer_faces.split_at_mut(j);
669            (&mut left[i], &mut right[0])
670        } else {
671            let (left, right) = block_outer_faces.split_at_mut(i);
672            (&mut right[0], &mut left[j])
673        };
674        let mut match_points =
675            find_matching_blocks(&blocks[i], &blocks[j], left, right, DEFAULT_TOL);
676        for points in match_points.drain(..) {
677            let mut face1 = create_face_from_diagonals(
678                &blocks[i],
679                points.iter().map(|p| p.i1).min().unwrap(),
680                points.iter().map(|p| p.j1).min().unwrap(),
681                points.iter().map(|p| p.k1).min().unwrap(),
682                points.iter().map(|p| p.i1).max().unwrap(),
683                points.iter().map(|p| p.j1).max().unwrap(),
684                points.iter().map(|p| p.k1).max().unwrap(),
685            );
686            face1.set_block_index(i);
687            let mut face2 = create_face_from_diagonals(
688                &blocks[j],
689                points.iter().map(|p| p.i2).min().unwrap(),
690                points.iter().map(|p| p.j2).min().unwrap(),
691                points.iter().map(|p| p.k2).min().unwrap(),
692                points.iter().map(|p| p.i2).max().unwrap(),
693                points.iter().map(|p| p.j2).max().unwrap(),
694                points.iter().map(|p| p.k2).max().unwrap(),
695            );
696            face2.set_block_index(j);
697            matches_to_remove.insert(face1.index_key());
698            matches_to_remove.insert(face2.index_key());
699
700            let corner1 = FaceRecord::from_match_points(i, &points, true).unwrap();
701            let corner2 = FaceRecord::from_match_points(j, &points, false).unwrap();
702            matches.push(FaceMatch {
703                block1: corner1,
704                block2: corner2,
705                points,
706            });
707        }
708    }
709
710    let mut outer_faces = Vec::new();
711    for faces in &block_outer_faces {
712        for face in faces {
713            outer_faces.push(face.clone());
714        }
715    }
716    let mut seen = HashSet::new();
717    outer_faces.retain(|face| seen.insert(face.index_key()));
718
719    outer_faces.retain(|face| !matches_to_remove.contains(&face.index_key()));
720
721    let mut outer_faces_to_remove = HashSet::new();
722    let mut by_block: HashMap<usize, Vec<&Face>> = HashMap::new();
723    for face in &outer_faces {
724        if let Some(idx) = face.block_index() {
725            by_block.entry(idx).or_default().push(face);
726        }
727    }
728    for faces in by_block.values() {
729        for (a_idx, face_a) in faces.iter().enumerate() {
730            let dims_a = [
731                face_a.imin(),
732                face_a.jmin(),
733                face_a.kmin(),
734                face_a.imax(),
735                face_a.jmax(),
736                face_a.kmax(),
737            ];
738            for (b_idx, face_b) in faces.iter().enumerate() {
739                if a_idx == b_idx {
740                    continue;
741                }
742                let dims_b = [
743                    face_b.imin(),
744                    face_b.jmin(),
745                    face_b.kmin(),
746                    face_b.imax(),
747                    face_b.jmax(),
748                    face_b.kmax(),
749                ];
750                let equal_components = dims_a
751                    .iter()
752                    .zip(dims_b.iter())
753                    .filter(|(a, b)| a == b)
754                    .count();
755                if equal_components == 5 {
756                    let remove_key = if face_b.diagonal_length() > face_a.diagonal_length() {
757                        face_b.index_key()
758                    } else {
759                        face_a.index_key()
760                    };
761                    outer_faces_to_remove.insert(remove_key);
762                }
763            }
764        }
765    }
766
767    outer_faces.retain(|face| !outer_faces_to_remove.contains(&face.index_key()));
768
769    for (idx, block) in blocks.iter().enumerate() {
770        let (_, self_matches) = get_outer_faces(block);
771        for (face_a, face_b) in self_matches {
772            let mut corner1 = FaceRecord {
773                block_index: idx,
774                imin: face_a.imin(),
775                jmin: face_a.jmin(),
776                kmin: face_a.kmin(),
777                imax: face_a.imax(),
778                jmax: face_a.jmax(),
779                kmax: face_a.kmax(),
780                id: face_a.id(),
781            };
782            let corner2 = FaceRecord {
783                block_index: idx,
784                imin: face_b.imin(),
785                jmin: face_b.jmin(),
786                kmin: face_b.kmin(),
787                imax: face_b.imax(),
788                jmax: face_b.jmax(),
789                kmax: face_b.kmax(),
790                id: face_b.id(),
791            };
792            corner1.id = face_a.id();
793            matches.push(FaceMatch {
794                block1: corner1,
795                block2: corner2,
796                points: Vec::new(),
797            });
798        }
799    }
800
801    let mut formatted = Vec::new();
802    let mut id_counter = 1;
803    for face in outer_faces {
804        formatted.push(FaceRecord {
805            block_index: face.block_index().unwrap_or(usize::MAX),
806            imin: face.imin(),
807            jmin: face.jmin(),
808            kmin: face.kmin(),
809            imax: face.imax(),
810            jmax: face.jmax(),
811            kmax: face.kmax(),
812            id: Some(id_counter),
813        });
814        id_counter += 1;
815    }
816
817    (matches, formatted)
818}

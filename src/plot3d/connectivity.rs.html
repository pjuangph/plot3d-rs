14/// Fields ending in `1` correspond to the first block/face, while those ending

28/// Compact record describing a face on a particular block.
29#[derive(Clone, Debug, Serialize)]
30pub struct FaceRecord {
31    pub block_index: usize,
32    pub imin: usize,
33    pub jmin: usize,
34    pub kmin: usize,
35    pub imax: usize,
36    pub jmax: usize,
37    pub kmax: usize,
38    pub id: Option<usize>,
39}
40
41impl FaceRecord {
42    /// Build a corner description from matching points.
43    ///
44    /// * `block_index` – Owning block index.
45    /// * `points` – Matched nodes.
46    /// * `first` – If `true` we use the indices from block1; otherwise block2.
47    ///
48    /// Returns `None` when `points` is empty.
49    fn from_match_points(block_index: usize, points: &[MatchPoint], first: bool) -> Option<Self> {
50        if points.is_empty() {
51            return None;
52        }
53        let imin = points
54            .iter()
55            .map(|p| if first { p.i1 } else { p.i2 })
56            .min()?;
57        let jmin = points
58            .iter()
59            .map(|p| if first { p.j1 } else { p.j2 })
60            .min()?;
61        let kmin = points
62            .iter()
63            .map(|p| if first { p.k1 } else { p.k2 })
64            .min()?;
65        let imax = points
66            .iter()
67            .map(|p| if first { p.i1 } else { p.i2 })
68            .max()?;
69        let jmax = points
70            .iter()
71            .map(|p| if first { p.j1 } else { p.j2 })
72            .max()?;
73        let kmax = points
74            .iter()
75            .map(|p| if first { p.k1 } else { p.k2 })
76            .max()?;
77        Some(Self {
78            block_index,
79            imin,
80            jmin,
81            kmin,
82            imax,
83            jmax,
84            kmax,
85            id: None,
86        })
87    }
88
89    /// Construct a record from a Face instance.
90    pub fn from_face(face: &crate::block_face_functions::Face) -> Self {
91        Self {
92            block_index: face.block_index().unwrap_or(usize::MAX),
93            imin: face.imin(),
94            jmin: face.jmin(),
95            kmin: face.kmin(),
96            imax: face.imax(),
97            jmax: face.jmax(),
98            kmax: face.kmax(),
99            id: face.id(),
100        }
101    }
102
103    /// Scale the index ranges by `factor`.
104    pub fn scale_indices(&mut self, factor: usize) {
105        if factor <= 1 {
106            return;
107        }
108        self.imin *= factor;
109        self.jmin *= factor;
110        self.kmin *= factor;
111        self.imax *= factor;
112        self.jmax *= factor;
113        self.kmax *= factor;
114    }
115
116    /// Reduce the index ranges by `divisor`.
117    pub fn divide_indices(&mut self, divisor: usize) {
118        if divisor <= 1 {
119            return;
120        }
121        self.imin /= divisor;
122        self.jmin /= divisor;
123        self.kmin /= divisor;
124        self.imax /= divisor;
125        self.jmax /= divisor;
126        self.kmax /= divisor;
127    }
128
129    /// Reconstruct a Face from this record using the provided blocks.
130    pub fn to_face(
131        &self,
132        blocks: &[crate::block::Block],
133    ) -> Option<crate::block_face_functions::Face> {
134        let block = blocks.get(self.block_index)?;
135        let mut face = crate::block_face_functions::create_face_from_diagonals(
136            block, self.imin, self.jmin, self.kmin, self.imax, self.jmax, self.kmax,
137        );
138        face.set_block_index(self.block_index);
139        if let Some(id) = self.id {
140            face.set_id(id);
141        }
142        Some(face)
143    }
144}

241/// Structured-grid node on a face, capturing indices and XYZ coordinate.
242#[derive(Clone, Debug)]
243struct FaceNode {
244    i: usize,
245    j: usize,
246    k: usize,
247    coord: [f64; 3],
248}
249
250/// Enumerate all nodes that belong to `face` on `block`.
251///
252/// # Arguments
253/// * `face` - Face whose nodes should be sampled.
254/// * `block` - Parent block providing Cartesian coordinates.
255///
256/// # Returns
257/// Vector of [`FaceNode`] containing structured indices `(i, j, k)` and the
258/// corresponding XYZ coordinate.
259fn face_nodes(face: &Face, block: &Block) -> Vec<FaceNode> {
260    let mut nodes = Vec::new();
261    let i_vals: Vec<usize> = if face.imin() == face.imax() {
262        vec![face.imin()]
263    } else {
264        (face.imin()..=face.imax()).collect()
265    };
266    let j_vals: Vec<usize> = if face.jmin() == face.jmax() {
267        vec![face.jmin()]
268    } else {
269        (face.jmin()..=face.jmax()).collect()
270    };
271    let k_vals: Vec<usize> = if face.kmin() == face.kmax() {
272        vec![face.kmin()]
273    } else {
274        (face.kmin()..=face.kmax()).collect()
275    };
276    for &i in &i_vals {
277        for &j in &j_vals {
278            for &k in &k_vals {
279                if !(i < block.imax && j < block.jmax && k < block.kmax) {
280                    continue;
281                }
282                let (x, y, z) = block.xyz(i, j, k);
283                nodes.push(FaceNode {
284                    i,
285                    j,
286                    k,
287                    coord: [x, y, z],
288                });
289            }
290        }
291    }
292    nodes
293}

295/// Locate the node whose coordinate is within `tol` of `target`.
296///
297/// Returns the first node that meets the tolerance, preferring the closest
298/// distance. When no node matches, `None` is returned.
299fn find_closest_node<'a>(
300    nodes: &'a [FaceNode],
301    target: [f64; 3],
302    tol: f64,
303) -> Option<&'a FaceNode> {
304    let mut best: Option<(&FaceNode, f64)> = None;
305    for node in nodes {
306        let dx = node.coord[0] - target[0];
307        let dy = node.coord[1] - target[1];
308        let dz = node.coord[2] - target[2];
309        let dist = (dx * dx + dy * dy + dz * dz).sqrt();
310        if dist <= tol {
311            match best {
312                Some((_, best_dist)) if dist >= best_dist => {}
313                _ => best = Some((node, dist)),
314            }
315        }
316    }
317    best.map(|(node, _)| node)
318}

320/// Check whether the coincident nodes degenerate to an edge contact.
321fn is_edge(points: &[MatchPoint]) -> bool {
322    if points.is_empty() {
323        return false;
324    }
325    let min_i1 = points.iter().map(|p| p.i1).min().unwrap();
326    let max_i1 = points.iter().map(|p| p.i1).max().unwrap();
327    let min_j1 = points.iter().map(|p| p.j1).min().unwrap();
328    let max_j1 = points.iter().map(|p| p.j1).max().unwrap();
329    let min_k1 = points.iter().map(|p| p.k1).min().unwrap();
330    let max_k1 = points.iter().map(|p| p.k1).max().unwrap();
331
332    let mut edge_matches = 0;
333    if min_i1 == max_i1 {
334        edge_matches += 1;
335    }
336    if min_j1 == max_j1 {
337        edge_matches += 1;
338    }
339    if min_k1 == max_k1 {
340        edge_matches += 1;
341    }
342    edge_matches >= 2
343}

345/// Filter matches so the provided key advances monotonically by 1.
346fn filter_block_increasing(
347    points: &[MatchPoint],
348    key: fn(&MatchPoint) -> usize,
349) -> Vec<MatchPoint> {
350    if points.is_empty() {
351        return Vec::new();
352    }
353    let mut unique_vals: Vec<usize> = points.iter().map(key).collect();
354    unique_vals.sort_unstable();
355    unique_vals.dedup();
356    if unique_vals.len() <= 1 {
357        return Vec::new();
358    }
359    let mut keep: HashSet<usize> = HashSet::new();
360    for window in unique_vals.windows(2) {
361        if window[1] == window[0] + 1 {
362            keep.insert(window[0]);
363        }
364    }
365    if let (Some(last), Some(prev)) = (
366        unique_vals.last(),
367        unique_vals.get(unique_vals.len().saturating_sub(2)),
368    ) {
369        if *last == *prev + 1 {
370            keep.insert(*last);
371        }
372    }
373    points
374        .iter()
375        .filter(|p| keep.contains(&key(p)))
376        .cloned()
377        .collect()
378}

380/// Enforce monotonic progression along the non-constant axes of each face.
381fn apply_axis_filters(points: Vec<MatchPoint>, face1: &Face, face2: &Face) -> Vec<MatchPoint> {
382    let mut filtered = points;
383    match face1.const_axis() {
384        Some(crate::block_face_functions::FaceAxis::I) => {
385            filtered = filter_block_increasing(&filtered, |p| p.j1);
386            filtered = filter_block_increasing(&filtered, |p| p.k1);
387        }
388        Some(crate::block_face_functions::FaceAxis::J) => {
389            filtered = filter_block_increasing(&filtered, |p| p.i1);
390            filtered = filter_block_increasing(&filtered, |p| p.k1);
391        }
392        Some(crate::block_face_functions::FaceAxis::K) => {
393            filtered = filter_block_increasing(&filtered, |p| p.i1);
394            filtered = filter_block_increasing(&filtered, |p| p.j1);
395        }
396        None => {}
397    }
398    match face2.const_axis() {
399        Some(crate::block_face_functions::FaceAxis::I) => {
400            filtered = filter_block_increasing(&filtered, |p| p.j2);
401            filtered = filter_block_increasing(&filtered, |p| p.k2);
402        }
403        Some(crate::block_face_functions::FaceAxis::J) => {
404            filtered = filter_block_increasing(&filtered, |p| p.i2);
405            filtered = filter_block_increasing(&filtered, |p| p.k2);
406        }
407        Some(crate::block_face_functions::FaceAxis::K) => {
408            filtered = filter_block_increasing(&filtered, |p| p.i2);
409            filtered = filter_block_increasing(&filtered, |p| p.j2);
410        }
411        None => {}
412    }
413    filtered
414}

416/// Build subfaces produced by the intersection region.
417fn create_split_faces(
418    face: &Face,
419    block: &Block,
420    points: &[MatchPoint],
421    use_block1: bool,
422) -> Vec<Face> {
423    if points.is_empty() {
424        return Vec::new();
425    }
426    let (imin, imax, jmin, jmax, kmin, kmax) = if use_block1 {
427        (
428            points.iter().map(|p| p.i1).min().unwrap(),
429            points.iter().map(|p| p.i1).max().unwrap(),
430            points.iter().map(|p| p.j1).min().unwrap(),
431            points.iter().map(|p| p.j1).max().unwrap(),
432            points.iter().map(|p| p.k1).min().unwrap(),
433            points.iter().map(|p| p.k1).max().unwrap(),
434        )
435    } else {
436        (
437            points.iter().map(|p| p.i2).min().unwrap(),
438            points.iter().map(|p| p.i2).max().unwrap(),
439            points.iter().map(|p| p.j2).min().unwrap(),
440            points.iter().map(|p| p.j2).max().unwrap(),
441            points.iter().map(|p| p.k2).min().unwrap(),
442            points.iter().map(|p| p.k2).max().unwrap(),
443        )
444    };
445    let degeneracy =
446        usize::from(imin == imax) + usize::from(jmin == jmax) + usize::from(kmin == kmax);
447    if degeneracy != 1 {
448        return Vec::new();
449    }
450    let mut split = split_face(face, block, imin, jmin, kmin, imax, jmax, kmax);
451    for f in &mut split {
452        if let Some(idx) = face.block_index() {
453            f.set_block_index(idx);
454        }
455        if let Some(id) = face.id() {
456            f.set_id(id);
457        }
458    }
459    split
460}

462/// Compute the coincident nodes between two faces on separate blocks.
463///
464/// # Arguments
465/// * `face1` - Candidate face on `block1`.
466/// * `face2` - Candidate face on `block2`.
467/// * `block1` / `block2` - Parent blocks.
468/// * `tol` - Euclidean tolerance for node matching.
469///
470/// # Returns
471/// Tuple containing:
472/// 1. List of [`MatchPoint`]s.
473/// 2. Split faces generated on `block1`.
474/// 3. Split faces generated on `block2`.
475pub fn get_face_intersection(
476    face1: &Face,
477    face2: &Face,
478    block1: &Block,
479    block2: &Block,
480    tol: f64,
481) -> (Vec<MatchPoint>, Vec<Face>, Vec<Face>) {
482    let nodes1 = face_nodes(face1, block1);
483    let nodes2 = face_nodes(face2, block2);
484    let mut matches = Vec::new();
485    for node1 in &nodes1 {
486        if let Some(node2) = find_closest_node(&nodes2, node1.coord, tol) {
487            matches.push(MatchPoint {
488                i1: node1.i,
489                j1: node1.j,
490                k1: node1.k,
491                i2: node2.i,
492                j2: node2.j,
493                k2: node2.k,
494            });
495        }
496    }
497    if matches.len() < 4 || is_edge(&matches) {
498        return (Vec::new(), Vec::new(), Vec::new());
499    }
500    let matches = apply_axis_filters(matches, face1, face2);
501    if matches.len() < 4 {
502        return (Vec::new(), Vec::new(), Vec::new());
503    }
504
505    let split_faces1 = create_split_faces(face1, block1, &matches, true);
506    let split_faces2 = create_split_faces(face2, block2, &matches, false);
507    (matches, split_faces1, split_faces2)
508}

510/// Recursively match all faces between a pair of blocks.
511///
512/// # Arguments
513/// * `block1` / `block2` - Blocks to compare.
514/// * `block1_outer` / `block2_outer` - Mutable outer-face lists that will be
515///   updated in-place as faces are split.
516/// * `tol` - Node matching tolerance.
517///
518/// # Returns
519/// Collection of match-point arrays, one entry per detected interface.
520pub fn find_matching_blocks(
521    block1: &Block,
522    block2: &Block,
523    block1_outer: &mut Vec<Face>,
524    block2_outer: &mut Vec<Face>,
525    tol: f64,
526) -> Vec<Vec<MatchPoint>> {
527    let mut matches = Vec::new();
528    let mut i = 0;
529    'outer: while i < block1_outer.len() {
530        let mut j = 0;
531        while j < block2_outer.len() {
532            let face1 = block1_outer[i].clone();
533            let face2 = block2_outer[j].clone();
534            let (match_points, split1, split2) =
535                get_face_intersection(&face1, &face2, block1, block2, tol);
536            if !match_points.is_empty() {
537                matches.push(match_points.clone());
538
539                block1_outer.remove(i);
540                block2_outer.remove(j);
541                block1_outer.extend(split1);
542                block2_outer.extend(split2);
543                i = 0;
544                continue 'outer;
545            } else {
546                j += 1;
547            }
548        }
549        i += 1;
550    }
551    matches
552}

554/// Generate block index pairs using centroid distance ordering.
555///
556/// # Arguments
557/// * `blocks` - All blocks in the assembly.
558/// * `nearest_nblocks` - Number of closest neighbours to include for each block.
559///
560/// # Returns
561/// Vector of ordered `(i, j)` index pairs.
562fn combinations_of_nearest_blocks(blocks: &[Block], nearest_nblocks: usize) -> Vec<(usize, usize)> {
563    let centroids: Vec<(f64, f64, f64)> = blocks.iter().map(Block::centroid).collect();
564    let mut combos = Vec::new();
565    for (i, &ci) in centroids.iter().enumerate() {
566        let mut distances: Vec<(usize, f64)> = centroids
567            .iter()
568            .enumerate()
569            .filter(|(j, _)| *j != i)
570            .map(|(j, cj)| {
571                let dx = ci.0 - cj.0;
572                let dy = ci.1 - cj.1;
573                let dz = ci.2 - cj.2;
574                (j, (dx * dx + dy * dy + dz * dz).sqrt())
575            })
576            .collect();
577        distances.sort_by(|a, b| a.1.partial_cmp(&b.1).unwrap());
578        for (j, dist) in distances.into_iter().take(nearest_nblocks) {
579            if dist.is_finite() {
580                combos.push((i, j));
581            }
582        }
583    }
584    combos
585}

587/// Connectivity computation performed on GCD-reduced blocks.
588///
589/// # Arguments
590/// * `blocks` - Original block list. Each block is down-sampled by the
591///   smallest index GCD across the set.
592///
593/// # Returns
594/// Tuple `(matches, outer_faces)` where `matches` enumerates face interfaces
595/// and `outer_faces` records the remaining external surfaces at the original
596/// resolution.
597pub fn connectivity_fast(blocks: &[Block]) -> (Vec<FaceMatch>, Vec<FaceRecord>) {
598    let mut gcd_array = Vec::with_capacity(blocks.len());
599    for block in blocks {
600        let gcd = gcd_three(block.imax - 1, block.jmax - 1, block.kmax - 1);
601        gcd_array.push(gcd);
602    }
603    let gcd_to_use = gcd_array.into_iter().min().unwrap_or(1).max(1);
604    let reduced_blocks = crate::block_face_functions::reduce_blocks(blocks, gcd_to_use);
605    let (mut matches, mut outer_faces) = connectivity(&reduced_blocks);
606    // Scale back to origional size
607    for face in &mut matches {
608        face.block1.imin *= gcd_to_use;
609        face.block1.jmin *= gcd_to_use;
610        face.block1.kmin *= gcd_to_use;
611        face.block1.imax *= gcd_to_use;
612        face.block1.jmax *= gcd_to_use;
613        face.block1.kmax *= gcd_to_use;
614
615        face.block2.imin *= gcd_to_use;
616        face.block2.jmin *= gcd_to_use;
617        face.block2.kmin *= gcd_to_use;
618        face.block2.imax *= gcd_to_use;
619        face.block2.jmax *= gcd_to_use;
620        face.block2.kmax *= gcd_to_use;
621    }
622    for face in &mut outer_faces {
623        face.imin *= gcd_to_use;
624        face.jmin *= gcd_to_use;
625        face.kmin *= gcd_to_use;
626        face.imax *= gcd_to_use;
627        face.jmax *= gcd_to_use;
628        face.kmax *= gcd_to_use;
629    }
630    (matches, outer_faces)
631}

633/// Determine face-to-face connectivity and exterior faces for all blocks.
634///
635/// # Arguments
636/// * `blocks` - Full-resolution blocks to analyse.
637///
638/// # Returns
639/// Tuple `(matches, outer_faces)` representing matched interfaces and the
640/// formatted list of outer faces.
641pub fn connectivity(blocks: &[Block]) -> (Vec<FaceMatch>, Vec<FaceRecord>) {
642    let mut block_outer_faces: Vec<Vec<Face>> = blocks
643        .iter()
644        .enumerate()
645        .map(|(idx, block)| {
646            let (faces, _) = get_outer_faces(block);
647            faces
648                .into_iter()
649                .map(|mut f| {
650                    f.set_block_index(idx);
651                    f
652                })
653                .collect()
654        })
655        .collect();
656
657    let combos = combinations_of_nearest_blocks(blocks, 6);
658    let mut matches = Vec::new();
659    let mut matches_to_remove: HashSet<(usize, usize, usize, usize, usize, usize, usize)> =
660        HashSet::new();
661
662    for (i, j) in combos {
663        if i == j {
664            continue;
665        }
666        let (left, right) = if i < j {
667            let (left, right) = block_outer_faces.split_at_mut(j);
668            (&mut left[i], &mut right[0])
669        } else {
670            let (left, right) = block_outer_faces.split_at_mut(i);
671            (&mut right[0], &mut left[j])
672        };
673        let mut match_points =
674            find_matching_blocks(&blocks[i], &blocks[j], left, right, DEFAULT_TOL);
675        for points in match_points.drain(..) {
676            let mut face1 = create_face_from_diagonals(
677                &blocks[i],
678                points.iter().map(|p| p.i1).min().unwrap(),
679                points.iter().map(|p| p.j1).min().unwrap(),
680                points.iter().map(|p| p.k1).min().unwrap(),
681                points.iter().map(|p| p.i1).max().unwrap(),
682                points.iter().map(|p| p.j1).max().unwrap(),
683                points.iter().map(|p| p.k1).max().unwrap(),
684            );
685            face1.set_block_index(i);
686            let mut face2 = create_face_from_diagonals(
687                &blocks[j],
688                points.iter().map(|p| p.i2).min().unwrap(),
689                points.iter().map(|p| p.j2).min().unwrap(),
690                points.iter().map(|p| p.k2).min().unwrap(),
691                points.iter().map(|p| p.i2).max().unwrap(),
692                points.iter().map(|p| p.j2).max().unwrap(),
693                points.iter().map(|p| p.k2).max().unwrap(),
694            );
695            face2.set_block_index(j);
696            matches_to_remove.insert(face1.index_key());
697            matches_to_remove.insert(face2.index_key());
698
699            let corner1 = FaceRecord::from_match_points(i, &points, true).unwrap();
700            let corner2 = FaceRecord::from_match_points(j, &points, false).unwrap();
701            matches.push(FaceMatch {
702                block1: corner1,
703                block2: corner2,
704                points,
705            });
706        }
707    }
708
709    let mut outer_faces = Vec::new();
710    for faces in &block_outer_faces {
711        for face in faces {
712            outer_faces.push(face.clone());
713        }
714    }
715    let mut seen = HashSet::new();
716    outer_faces.retain(|face| seen.insert(face.index_key()));
717
718    outer_faces.retain(|face| !matches_to_remove.contains(&face.index_key()));
719
720    let mut outer_faces_to_remove = HashSet::new();
721    let mut by_block: HashMap<usize, Vec<&Face>> = HashMap::new();
722    for face in &outer_faces {
723        if let Some(idx) = face.block_index() {
724            by_block.entry(idx).or_default().push(face);
725        }
726    }
727    for faces in by_block.values() {
728        for (a_idx, face_a) in faces.iter().enumerate() {
729            let dims_a = [
730                face_a.imin(),
731                face_a.jmin(),
732                face_a.kmin(),
733                face_a.imax(),
734                face_a.jmax(),
735                face_a.kmax(),
736            ];
737            for (b_idx, face_b) in faces.iter().enumerate() {
738                if a_idx == b_idx {
739                    continue;
740                }
741                let dims_b = [
742                    face_b.imin(),
743                    face_b.jmin(),
744                    face_b.kmin(),
745                    face_b.imax(),
746                    face_b.jmax(),
747                    face_b.kmax(),
748                ];
749                let equal_components = dims_a
750                    .iter()
751                    .zip(dims_b.iter())
752                    .filter(|(a, b)| a == b)
753                    .count();
754                if equal_components == 5 {
755                    let remove_key = if face_b.diagonal_length() > face_a.diagonal_length() {
756                        face_b.index_key()
757                    } else {
758                        face_a.index_key()
759                    };
760                    outer_faces_to_remove.insert(remove_key);
761                }
762            }
763        }
764    }
765
766    outer_faces.retain(|face| !outer_faces_to_remove.contains(&face.index_key()));
767
768    for (idx, block) in blocks.iter().enumerate() {
769        let (_, self_matches) = get_outer_faces(block);
770        for (face_a, face_b) in self_matches {
771            let mut corner1 = FaceRecord {
772                block_index: idx,
773                imin: face_a.imin(),
774                jmin: face_a.jmin(),
775                kmin: face_a.kmin(),
776                imax: face_a.imax(),
777                jmax: face_a.jmax(),
778                kmax: face_a.kmax(),
779                id: face_a.id(),
780            };
781            let corner2 = FaceRecord {
782                block_index: idx,
783                imin: face_b.imin(),
784                jmin: face_b.jmin(),
785                kmin: face_b.kmin(),
786                imax: face_b.imax(),
787                jmax: face_b.jmax(),
788                kmax: face_b.kmax(),
789                id: face_b.id(),
790            };
791            corner1.id = face_a.id();
792            matches.push(FaceMatch {
793                block1: corner1,
794                block2: corner2,
795                points: Vec::new(),
796            });
797        }
798    }
799
800    let mut formatted = Vec::new();
801    let mut id_counter = 1;
802    for face in outer_faces {
803        formatted.push(FaceRecord {
804            block_index: face.block_index().unwrap_or(usize::MAX),
805            imin: face.imin(),
806            jmin: face.jmin(),
807            kmin: face.kmin(),
808            imax: face.imax(),
809            jmax: face.jmax(),
810            kmax: face.kmax(),
811            id: Some(id_counter),
812        });
813        id_counter += 1;
814    }
815
816    (matches, formatted)
817}

819/// Greatest common divisor of two integers.
820fn gcd_two(mut a: usize, mut b: usize) -> usize {
821    while b != 0 {
822        let r = a % b;
823        a = b;
824        b = r;
825    }
826    a
827}
828
829/// Greatest common divisor of three integers.
830fn gcd_three(a: usize, b: usize, c: usize) -> usize {
831    gcd_two(gcd_two(a, b), c)
832}

18pub fn write_plot3d(
19    path: &str,
20    blocks: &[Block],
21    binary: bool,
22    format: BinaryFormat,
23    precision: FloatPrecision,
24    endian: Endian,
25) -> std::io::Result<()> {
26    if binary {
27        let f = File::create(path)?;
28        let mut w = BufWriter::new(f);
29        match format {
30            BinaryFormat::Raw => write_raw(&mut w, blocks, precision, endian),
31            BinaryFormat::Fortran => write_fortran(&mut w, blocks, precision, endian),
32        }
33    } else {
34        write_ascii(path, blocks)
35    }
36}

38fn write_ascii(path: &str, blocks: &[Block]) -> std::io::Result<()> {
39    let mut w = BufWriter::new(File::create(path)?);
40    writeln!(w, "{}", blocks.len())?;
41    for b in blocks {
42        writeln!(w, "{} {} {}", b.imax, b.jmax, b.kmax)?;
43    }
44    for b in blocks {
45        write_var_ascii(&mut w, &b.x)?;
46        write_var_ascii(&mut w, &b.y)?;
47        write_var_ascii(&mut w, &b.z)?;
48    }
49    Ok(())
50}

52fn write_var_ascii(w: &mut impl Write, v: &[f64]) -> std::io::Result<()> {
53    let mut col = 0usize;
54    for val in v {
55        write!(w, "{:.8} ", val)?;
56        col += 1;
57        if col % 6 == 0 {
58            writeln!(w)?;
59        }
60    }
61    if col % 6 != 0 {
62        writeln!(w)?;
63    }
64    Ok(())
65}

67fn write_raw(
68    mut w: &mut impl Write,
69    blocks: &[Block],
70    precision: FloatPrecision,
71    endian: Endian,
72) -> std::io::Result<()> {
73    use byteorder::{BigEndian, LittleEndian, WriteBytesExt};
74
75    // header
76    match endian {
77        Endian::Little => w.write_u32::<LittleEndian>(blocks.len() as u32)?,
78        Endian::Big => w.write_u32::<BigEndian>(blocks.len() as u32)?,
79    }
80    for b in blocks {
81        match endian {
82            Endian::Little => {
83                w.write_u32::<LittleEndian>(b.imax as u32)?;
84                w.write_u32::<LittleEndian>(b.jmax as u32)?;
85                w.write_u32::<LittleEndian>(b.kmax as u32)?;
86            }
87            Endian::Big => {
88                w.write_u32::<BigEndian>(b.imax as u32)?;
89                w.write_u32::<BigEndian>(b.jmax as u32)?;
90                w.write_u32::<BigEndian>(b.kmax as u32)?;
91            }
92        }
93    }
94
95    // payload
96    for b in blocks {
97        write_vec_num(&mut w, &b.x, precision, endian)?;
98        write_vec_num(&mut w, &b.y, precision, endian)?;
99        write_vec_num(&mut w, &b.z, precision, endian)?;
100    }
101    Ok(())
102}

104fn write_fortran(
105    mut w: &mut impl Write,
106    blocks: &[Block],
107    precision: FloatPrecision,
108    endian: Endian,
109) -> std::io::Result<()> {
110    // header: [nblocks] as a record
111    let mut nb = [0u8; 4];
112    utils::Endian::write_u32(&mut nb, blocks.len() as u32, endian);
113    write_fortran_record(&mut w, &nb, endian)?;
114
115    // dims as one record per block
116    for b in blocks {
117        let mut rec = [0u8; 12];
118        utils::Endian::write_u32(&mut rec[0..4], b.imax as u32, endian);
119        utils::Endian::write_u32(&mut rec[4..8], b.jmax as u32, endian);
120        utils::Endian::write_u32(&mut rec[8..12], b.kmax as u32, endian);
121        write_fortran_record(&mut w, &rec, endian)?;
122    }
123
124    // payload: 3 records per block: X, Y, Z
125    for b in blocks {
126        match precision {
127            FloatPrecision::F32 => {
128                let xb = utils::Endian::write_f32_slice(
129                    &b.x.iter().map(|v| *v as f32).collect::<Vec<f32>>(),
130                    endian,
131                );
132                write_fortran_record(&mut w, &xb, endian)?;
133                let yb = utils::Endian::write_f32_slice(
134                    &b.y.iter().map(|v| *v as f32).collect::<Vec<f32>>(),
135                    endian,
136                );
137                write_fortran_record(&mut w, &yb, endian)?;
138                let zb = utils::Endian::write_f32_slice(
139                    &b.z.iter().map(|v| *v as f32).collect::<Vec<f32>>(),
140                    endian,
141                );
142                write_fortran_record(&mut w, &zb, endian)?;
143            }
144            FloatPrecision::F64 => {
145                let xb = utils::Endian::write_f64_slice(&b.x, endian);
146                write_fortran_record(&mut w, &xb, endian)?;
147                let yb = utils::Endian::write_f64_slice(&b.y, endian);
148                write_fortran_record(&mut w, &yb, endian)?;
149                let zb = utils::Endian::write_f64_slice(&b.z, endian);
150                write_fortran_record(&mut w, &zb, endian)?;
151            }
152        }
153    }
154
155    Ok(())
156}

158fn write_vec_num(
159    w: &mut impl Write,
160    v: &[f64],
161    precision: FloatPrecision,
162    endian: Endian,
163) -> std::io::Result<()> {
164    use byteorder::{BigEndian, LittleEndian, WriteBytesExt};
165    match (precision, endian) {
166        (FloatPrecision::F32, Endian::Little) => {
167            for &f in v {
168                w.write_f32::<LittleEndian>(f as f32)?;
169            }
170        }
171        (FloatPrecision::F32, Endian::Big) => {
172            for &f in v {
173                w.write_f32::<BigEndian>(f as f32)?;
174            }
175        }
176        (FloatPrecision::F64, Endian::Little) => {
177            for &f in v {
178                w.write_f64::<LittleEndian>(f)?;
179            }
180        }
181        (FloatPrecision::F64, Endian::Big) => {
182            for &f in v {
183                w.write_f64::<BigEndian>(f)?;
184            }
185        }
186    }
187    Ok(())
